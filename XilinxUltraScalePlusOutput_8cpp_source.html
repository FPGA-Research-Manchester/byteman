29 void XilinxUltraScalePlus::writeBitstream(string filename, string params, int srcRow, int srcCol, int sizeRow, int sizeCol){
30  size_t dotpos = filename.rfind(".");
31  if(dotpos == string::npos)
32  throw runtime_error(string("Invalid file name: \"").append(filename).append("\"!\n"));
33  designName = filename.substr(0, dotpos);
34 
35  if(verbose)cout<<"Writing Xilinx UltraScale+ bitstream to file \""<<filename<<"\":\n";
36  ofstream fout (filename, ofstream::binary | ofstream::trunc);
37  if(!fout.good())
38  throw runtime_error(string("Could not open file: \"").append(filename).append("\"!\n"));
39  if(fileExtensionIs(filename, "bit"))
40  writeBitstreamBIT(fout, params, srcRow, srcCol, sizeRow, sizeCol);
41  else
42  throw runtime_error(string("Unknown Xilinx UltraScale+ file format tried to be written.\n"));
43  fout.close();
44 }

59 void XilinxUltraScalePlus::writeBitstreamBIT(ofstream& fout, string params, int srcRow, int srcCol, int sizeRow, int sizeCol){
60 
61 
62  //Parse output params
63  int outputCLB = 0, outputBRAM = 0, blankCLBs = 0, partial = 1;
64  if(params.find("logic") != string::npos || params.find("clb") != string::npos)outputCLB++;
65  if(params.find("blockram") != string::npos || params.find("bram") != string::npos)outputBRAM++;
66  if(params.find("blank") != string::npos)blankCLBs++;
67  if(params.find("full") != string::npos)partial = 0;
68 
69  if(outputCLB == 0 && outputBRAM == 0){ // if nothing was chosen, choose both
70  outputCLB = outputBRAM = 1;
71  }
72  if(partial) {
73  if((srcRow%XUSP_CLB_PER_CLOCK_REGION != 0) || (sizeRow%XUSP_CLB_PER_CLOCK_REGION != 0))
74  throw runtime_error("Currently only full clock region height relocations are supported (use row numbers multiple of 60).");
75  if(sizeRow <= 0 || sizeCol <= 0)
76  throw runtime_error("Output size needs to be larger than 0.");
77  }
78  //for now overwrite the design name
79  if(partial)
80  designName.append(";PARTIAL=TRUE");
81  designName.append(";bytemanVersion=").append(VERSION).append(":").append(VERSION_BUILD);
82 
83  { //Update fileDate, fileTime with {now}
84  time_t timestamp = time(0);
85  struct tm tstruct;
86  char buf[80];
87  tstruct = *localtime(&timestamp);
88  strftime(buf, sizeof(buf), "%Y/%m/%d", &tstruct);
89  fileDate = string(buf);
90  strftime(buf, sizeof(buf), "%H:%M:%S", &tstruct);
91  fileTime = string(buf);
92  }
93  streamoff headerLocationOfRemainingFileLength = XilinxUltraScalePlus::outputBITheader(fout);
94  CAPwriteNOPs(fout, 2);
95 
96  //TODO, add support for multi-SLR
97  CAPwriteNOPs(fout, 132);
98  CAPwriteReg(fout, XUSP_CAP_REG_CMD, XUSP_CAP_CMD_RCRC);
99  CAPwriteNOPs(fout, 2);
100  CAPwriteReg(fout, XUSP_CAP_REG_IDCODE, SLRinfo[0].IDCODE);
101  CAPwriteReg(fout, XUSP_CAP_REG_CMD, XUSP_CAP_CMD_NULL);
102  CAPwriteReg(fout, XUSP_CAP_REG_MASK, 0x500);
103  CAPwriteReg(fout, XUSP_CAP_REG_CTRL0, 0x500);
104  CAPwriteReg(fout, XUSP_CAP_REG_MASK, 0x20000);
105  CAPwriteReg(fout, XUSP_CAP_REG_CTRL1, 0);
106  if(partial){
107  for (rect r : regionsSelected) {
108  XilinxUltraScalePlus::outputBITregion(fout, outputCLB, outputBRAM, blankCLBs, r.posRow, r.posCol, r.sizeRow, r.sizeCol);
109  }
110  XilinxUltraScalePlus::outputBITregion(fout, outputCLB, outputBRAM, blankCLBs, srcRow, srcCol, sizeRow, sizeCol);
111  } else {// full

113  }
114  XilinxUltraScalePlus::outputBITfooter(fout);
115  XilinxUltraScalePlus::outputBITheaderLengthField(fout, headerLocationOfRemainingFileLength);

117 streamoff XilinxUltraScalePlus::outputBITheader(ofstream& fout){
118  fileWriteBE(fout, 16, 9);//16-bit BE value = 9
119  fileWriteString(fout, string("\x0F\xF0\x0F\xF0\x0F\xF0\x0F\xF0"));
120  fileWriteBE(fout, 16, 1);//16-bit BE value = 1
121  fileWriteBE(fout, 8, 'a');
122  fileWriteBE(fout, 16, static_cast<uint16_t>(designName.size()+1));
123  fileWriteString(fout, designName);
124  fileWriteBE(fout, 8, 'b');//8-bit BE value = 'b'
125  fileWriteBE(fout, 16, static_cast<uint16_t>(partName.size()+1));
126  fileWriteString(fout, partName);
127  fileWriteBE(fout, 8, 'c');//8-bit BE value = 'c'
128  fileWriteBE(fout, 16, static_cast<uint16_t>(fileDate.size()+1));
129  fileWriteString(fout, fileDate);
130  fileWriteBE(fout, 8, 'd');//8-bit BE value = 'd'
131  fileWriteBE(fout, 16, static_cast<uint16_t>(fileTime.size()+1));
132  fileWriteString(fout, fileTime);
133  fileWriteBE(fout, 8, 'e');//8-bit BE value = 'e'
134  streamoff headerLocationOfRemainingFileLength = fout.tellp();
135  fileWriteBE(fout, 32, 0);//32-bit BE value = 0 representing remaining file size. needs to be fixed later
136  for(int i = 0 ; i < 16 ; i++) // 64 bytes of 0xFF
137  fileWriteBE(fout, 32, 0xFFFFFFFF);//32-bit BE value = 0xFFFFFFFF
138  fileWriteBE(fout, 32, 0x000000BB);
139  fileWriteBE(fout, 32, 0x11220044);
140  fileWriteBE(fout, 32, 0xFFFFFFFF);
141  fileWriteBE(fout, 32, 0xFFFFFFFF);
142 
143  CAPwriteSync(fout);
144  return headerLocationOfRemainingFileLength;
145 }
