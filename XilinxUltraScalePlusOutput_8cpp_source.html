38 void XilinxUltraScalePlus::writeBitstream(string filename, string params, Rect2D cmdRect)
39 {
40  size_t dotpos = filename.rfind(".");
41  if(dotpos == string::npos)
42  throw runtime_error(string("Invalid file name: \"").append(filename).append("\"!\n"));
43  designName = filename.substr(0, dotpos);
44 
45  log("Writing Xilinx UltraScale+ bitstream to file \"" + filename + "\":");
46 
47  SelectedOptions options = XilinxUltraScalePlus::parseParams(params);
48 
49  if(options.partial) {
50  if((cmdRect.position.row % XUSP_CLB_PER_CLOCK_REGION != 0) || (cmdRect.size.row % XUSP_CLB_PER_CLOCK_REGION != 0))
51  throw runtime_error("Currently only full clock region height relocations are supported (use row numbers multiple of 60).");
52  if(cmdRect.size.row <= 0 || cmdRect.size.col <= 0)
53  throw runtime_error("Invalid output size dimentions.");
54  }
55 
56  ofstream fout (filename, ofstream::binary | ofstream::trunc);
57  if(!fout.good())
58  throw runtime_error(string("Could not open file: \"").append(filename).append("\"!\n"));
59  if(str::iff::stringEndsWith(filename, ".bit"))
60  XilinxUltraScalePlus::writeBitstreamBIT(fout, params, cmdRect, options);
61  else if(str::iff::stringEndsWith(filename, ".bin"))
62  XilinxUltraScalePlus::writeBitstreamBIN(fout, params, cmdRect, options);
63  else
64  throw runtime_error(string("Unknown Xilinx UltraScale+ file format tried to be written.\n"));
65  fout.close();
66 }

84 void XilinxUltraScalePlus::writeBitstreamMain(ofstream& fout, string params, Rect2D cmdRect, SelectedOptions options)
85 {
86  streamoff slrMagicInstrLocation[XUSP_MAX_SLRS];
87 
88  for(int slr = 0 ; slr < numberOfSLRs ; slr++){
89  //Check if there is overlap between selected regions and the current SLR.
90  Rect2D slrCoordsRect = {{0, 0}, {0, 0}};
91  slrCoordsRect.position.row = SLRinfo[slr].fromRow * XUSP_CLB_PER_CLOCK_REGION;
92  slrCoordsRect.size.row = (SLRinfo[slr].toRow - SLRinfo[slr].fromRow + 1) * XUSP_CLB_PER_CLOCK_REGION;
93  slrCoordsRect.size.col = numberOfCols;
94  bool overlap = false;
95  if(options.partial){
96  for (Rect2D selRect : regionSelection) {
97  if(!rect::empty(rect::getOverlap(selRect, slrCoordsRect))){
98  overlap = true;
99  break;
100  }
101  }
102  if(!rect::empty(rect::getOverlap(cmdRect, slrCoordsRect)))
103  overlap = true;
104  } else {// full bitstream output
105  overlap = true;
106  }
107 
108  if(overlap)
109  slrMagicInstrLocation[slr] = XilinxUltraScalePlus::writeBitstreamMainSingleSLR(fout, slr, cmdRect, options);
110  else
111  slrMagicInstrLocation[slr] = XilinxUltraScalePlus::writeBitstreamMainEmptySLR(fout, slr, options);
112  }
113  for(int slr = numberOfSLRs - 2 ; slr >= 0 ; slr--){
114  { //fix magic command value
115  streamoff currentStreamHead = fout.tellp();
116  streamoff bypassSize = (currentStreamHead - slrMagicInstrLocation[slr])/4 - 1;
117  fout.seekp(slrMagicInstrLocation[slr]);
118  FileIO::write32(fout, static_cast<uint32_t>(bypassSize), loadedBitstreamEndianess);
119  fout.seekp(currentStreamHead);
120  }
121  XilinxUltraScalePlus::writeBitstreamMainSLRfooter(fout, slr, options);
122  }
123  XilinxUltraScalePlus::CAP_writeNOP(fout, 400, 0, loadedBitstreamEndianess);
124 }

127

133  XilinxUltraScalePlus::CAP_writeNOP(fout, 2, 0, loadedBitstreamEndianess);
134 
135  streamoff fileLocationOfSLRmagicWrite;
136  if((slr + 1) < numberOfSLRs) { // If there are more SLRs, output the magic
137  XilinxUltraScalePlus::CAP_writeSelectRegister(fout, CAP::Register::MAGIC1, loadedBitstreamEndianess);
138  fileLocationOfSLRmagicWrite = fout.tellp();
139  XilinxUltraScalePlus::CAP_writeType2(fout, 0, loadedBitstreamEndianess);
140  }
141  return fileLocationOfSLRmagicWrite;

214 void XilinxUltraScalePlus::writeBitstreamMainSingleRegion(ofstream& fout, int slr, Rect2D writeRect, SelectedOptions options)
215 {
216  if(rect::empty(writeRect)) // If output region is empty, just return
217  return;
218  int srcR = writeRect.position.row / XUSP_CLB_PER_CLOCK_REGION;
219  int sizeR = writeRect.size.row / XUSP_CLB_PER_CLOCK_REGION;
220  if(options.clb){
221  for(int r = 0 ; r < sizeR ; r++){
222  int framesToWrite = numberOfFramesBeforeCol[writeRect.position.col+writeRect.size.col]-numberOfFramesBeforeCol[writeRect.position.col];
223  uint32_t farValue = XilinxUltraScalePlus::CAP_makeFAR(static_cast<int>(CAP::BlockType::LOGIC), slr, srcR+r, writeRect.position.col, 0);
224  if(options.blank){
225  XilinxUltraScalePlus::CAP_writeRegister(fout, CAP::Register::FAR, farValue, loadedBitstreamEndianess);
226  XilinxUltraScalePlus::CAP_writeCommand(fout, CAP::Command::WCFG, loadedBitstreamEndianess);
227  XilinxUltraScalePlus::CAP_writeNOP(fout, 1, 0, loadedBitstreamEndianess);
228  XilinxUltraScalePlus::CAP_writeFDRI(fout, ((framesToWrite+1)*XUSP_WORDS_PER_FRAME), loadedBitstreamEndianess);
229  for(int i = 0 ; i <= framesToWrite ; i++)
230  fout.write((char*)blankFrame, XUSP_WORDS_PER_FRAME * 4);
231  }//options.blank
232  XilinxUltraScalePlus::CAP_writeRegister(fout, CAP::Register::FAR, farValue, loadedBitstreamEndianess);
233  XilinxUltraScalePlus::CAP_writeCommand(fout, CAP::Command::WCFG, loadedBitstreamEndianess);
234  XilinxUltraScalePlus::CAP_writeNOP(fout, 1, 0, loadedBitstreamEndianess);
235  XilinxUltraScalePlus::CAP_writeFDRI(fout, ((framesToWrite+1)*XUSP_WORDS_PER_FRAME), loadedBitstreamEndianess);
236 
237  fout.write((char*)&bitstreamCLB[srcR+r][writeRect.position.col][0], framesToWrite * XUSP_WORDS_PER_FRAME * 4);
238  fout.write((char*)blankFrame, XUSP_WORDS_PER_FRAME * 4);
239  }
240  }//options.clb
241  //TODO
242  if(options.bram){
243  for(int r = 0 ; r < sizeR ; r++){
244  int framesToWrite = XUSP_FRAMES_PER_BRAM_CONTENT_COLUMN * (numberOfBRAMsBeforeCol[writeRect.position.col+writeRect.size.col]-numberOfBRAMsBeforeCol[writeRect.position.col]);
245  if(framesToWrite > 0) {
246  uint32_t farValue = XilinxUltraScalePlus::CAP_makeFAR(static_cast<int>(CAP::BlockType::BLOCKRAM), slr, srcR+r, numberOfBRAMsBeforeCol[writeRect.position.col], 0);
247  XilinxUltraScalePlus::CAP_writeRegister(fout, CAP::Register::FAR, farValue, loadedBitstreamEndianess);
248  XilinxUltraScalePlus::CAP_writeCommand(fout, CAP::Command::WCFG, loadedBitstreamEndianess);
249  XilinxUltraScalePlus::CAP_writeNOP(fout, 1, 0, loadedBitstreamEndianess);
250  XilinxUltraScalePlus::CAP_writeFDRI(fout, ((framesToWrite+1)*XUSP_WORDS_PER_FRAME), loadedBitstreamEndianess);
251 
252  fout.write((char*)&bitstreamBRAM[srcR+r][numberOfBRAMsBeforeCol[writeRect.position.col]][0], framesToWrite * XUSP_WORDS_PER_FRAME * 4);
253  fout.write((char*)blankFrame, XUSP_WORDS_PER_FRAME * 4);
254  }
255  }
256  }//options.blockram
