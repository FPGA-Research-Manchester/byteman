38 void XilinxUltraScalePlus::writeBitstream(string filename, string params, Rect2D rect){
39  size_t dotpos = filename.rfind(".");
40  if(dotpos == string::npos)
41  throw runtime_error(string("Invalid file name: \"").append(filename).append("\"!\n"));
42  designName = filename.substr(0, dotpos);
43 
44  if(verbose)cout<<"Writing Xilinx UltraScale+ bitstream to file \""<<filename<<"\":\n";
45 
46  SelectedOptions options = XilinxUltraScalePlus::parseParams(params);
47 
48  if(options.partial) {
49  if((rect.position.row % XUSP_CLB_PER_CLOCK_REGION != 0) || (rect.size.row % XUSP_CLB_PER_CLOCK_REGION != 0))
50  throw runtime_error("Currently only full clock region height relocations are supported (use row numbers multiple of 60).");
51  if(rect.size.row <= 0 || rect.size.col <= 0)
52  throw runtime_error("Invalid output size dimentions.");
53  }
54 
55  ofstream fout (filename, ofstream::binary | ofstream::trunc);
56  if(!fout.good())
57  throw runtime_error(string("Could not open file: \"").append(filename).append("\"!\n"));
58  if(StringFuncs::checkIf::stringEndsWith(filename, ".bit"))
59  XilinxUltraScalePlus::writeBitstreamBIT(fout, params, rect, options);
60  else if(StringFuncs::checkIf::stringEndsWith(filename, ".bin"))
61  XilinxUltraScalePlus::writeBitstreamBIN(fout, params, rect, options);
62  else
63  throw runtime_error(string("Unknown Xilinx UltraScale+ file format tried to be written.\n"));
64  fout.close();
65 }

123 void XilinxUltraScalePlus::writeBitstreamMainSingleRegion(ofstream& fout, Rect2D rect, SelectedOptions options){
124  int srcR = rect.position.row / XUSP_CLB_PER_CLOCK_REGION;
125  int sizeR = rect.size.row / XUSP_CLB_PER_CLOCK_REGION;
126  if(options.clb){
127  for(int r = 0 ; r < sizeR ; r++){
128  int framesToWrite = numberOfFramesBeforeCol[rect.position.col+rect.size.col]-numberOfFramesBeforeCol[rect.position.col];
129  uint32_t farValue = XilinxUltraScalePlus::CAP_makeFAR(static_cast<int>(CAP::BlockType::LOGIC), srcR+r, rect.position.col, 0);
130  if(options.blank){
131  XilinxUltraScalePlus::CAP_writeRegister(fout, CAP::Register::FAR, farValue, loadedBitstreamEndianess);
132  XilinxUltraScalePlus::CAP_writeCommand(fout, CAP::Command::WCFG, loadedBitstreamEndianess);
133  XilinxUltraScalePlus::CAP_writeNOP(fout, 1, 0, loadedBitstreamEndianess);
134  XilinxUltraScalePlus::CAP_writeFDRI(fout, ((framesToWrite+1)*XUSP_WORDS_PER_FRAME), loadedBitstreamEndianess);
135  for(int i = 0 ; i <= framesToWrite ; i++)
136  fout.write((char*)blankFrame, XUSP_WORDS_PER_FRAME * 4);
137  }//options.blank
138  XilinxUltraScalePlus::CAP_writeRegister(fout, CAP::Register::FAR, farValue, loadedBitstreamEndianess);
139  XilinxUltraScalePlus::CAP_writeCommand(fout, CAP::Command::WCFG, loadedBitstreamEndianess);
140  XilinxUltraScalePlus::CAP_writeNOP(fout, 1, 0, loadedBitstreamEndianess);
141  XilinxUltraScalePlus::CAP_writeFDRI(fout, ((framesToWrite+1)*XUSP_WORDS_PER_FRAME), loadedBitstreamEndianess);
142 
143  fout.write((char*)&bitstreamCLB[srcR+r][rect.position.col][0], framesToWrite * XUSP_WORDS_PER_FRAME * 4);
