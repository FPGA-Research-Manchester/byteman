164 void parseCommand(string nextCmd){
165  replace(nextCmd.begin(), nextCmd.end(), '=', ' ');
166  replace(nextCmd.begin(), nextCmd.end(), ':', ' ');
167  if (nextCmd.at(0) == '-') nextCmd.erase(0, 1);
168  #define cmdIs(x,y) (0 == nextCmd.rfind(x, 0))
169  if(cmdIs("h", "help")){
170  parseHelp(nextCmd);
171  return;
172  }
173  if(!archSelected)setArch(nextCmd);
174  if(cmdIs("v", "verbose"))
175  parseVerbose(nextCmd);
176  else if(cmdIs("w", "warn"))
177  parseWarn(nextCmd);
178  else if(cmdIs("r", "region"))
179  parseRegion(nextCmd);
180  else if(cmdIs("b", "blank"))
181  parseBlank(nextCmd);
182  else if(cmdIs("c", "change"))
183  parseChange(nextCmd);
184  else if(cmdIs("d", "device"))
185  parseDevice(nextCmd);
186  else if(cmdIs("i", "input"))
187  parseInput(nextCmd);
188  else if(cmdIs("m", "merge"))
189  parseMerge(nextCmd);
190  else if(cmdIs("o", "output"))
191  parseOutput(nextCmd);
192  else if(cmdIs("a", "assembly"))
193  parseAssembly(nextCmd);
194  else if(cmdIs("e", "end"))
195  exit(0);
196  #undef cmdIs
197 }
