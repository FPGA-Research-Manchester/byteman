239 void setArch(string arch){
240  archSelected = true; // it will either select it or throw an exception, so just mark it anyway who cares
241  #define has(x) (arch.rfind(x) != string::npos)
242  #define hOR(x,y) (has(x)||has(y))
243  transform(arch.begin(), arch.end(),arch.begin(), ::toupper);
244  #ifdef XUSP
245  if(has("XUSP") || (has("XILINX") && hOR("ULTRASCALE", "US") && hOR("PLUS","+") ) ) {
246  selectedArch = XIL_USP;
247  return;
248  }
249  #endif
250  #undef has
251  #undef hOR
252 
253  throw runtime_error(string("Unknown architecture: \"").append(arch).append("\"."));
254 }
255 void parseCommand(string nextCmd){
256  replace(nextCmd.begin(), nextCmd.end(), '=', ' ');
257  if (nextCmd.at(0) == '-') nextCmd.erase(0, 1);
258  #define cmdIs(x,y) (nextCmd.rfind(x, 0) == 0)
259  if(cmdIs("h", "help")){
260  parseHelp(nextCmd);
261  return;
262  }
263  if(!archSelected)setArch(nextCmd);
264  if(cmdIs("v", "verbose"))
265  parseVerbose(nextCmd);
266  else if(cmdIs("w", "warn"))
267  parseWarn(nextCmd);
268  else if(cmdIs("r", "region"))
269  parseRegion(nextCmd);
270  else if(cmdIs("b", "blank"))
271  parseBlank(nextCmd);
272  else if(cmdIs("c", "change"))
273  parseChange(nextCmd);
274  else if(cmdIs("d", "device"))
275  parseDevice(nextCmd);
276  else if(cmdIs("i", "input"))
277  parseInput(nextCmd);
278  else if(cmdIs("m", "merge"))
279  parseMerge(nextCmd);
280  else if(cmdIs("o", "output"))
281  parseOutput(nextCmd);
282  else if(cmdIs("a", "assembly"))
283  parseAssembly(nextCmd);
284  else if(cmdIs("e", "end"))
285  exit(0);
286  #undef cmdIs
287 }
