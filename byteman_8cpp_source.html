163 void setArch(string arch){
164  archSelected = true; // it will either select it or throw an exception, so just mark it anyway who cares
165  #define has(x) (arch.rfind(x) != string::npos)
166  #define hOR(x,y) (has(x)||has(y))
167  transform(arch.begin(), arch.end(),arch.begin(), ::toupper);
168  #ifdef XUSP
169  if(has("XUSP") || (has("XILINX") && hOR("ULTRASCALE", "US") && hOR("PLUS","+") ) ) {
170  selectedArch = XIL_USP;
171  return;
172  }
173  #endif
174  #undef has
175  #undef hOR
176 
177  throw runtime_error(string("Unknown architecture: \"").append(arch).append("\"."));
178 }
179 void parseCommand(string nextCmd){
180  replace(nextCmd.begin(), nextCmd.end(), '=', ' ');
181  if (nextCmd.at(0) == '-') nextCmd.erase(0, 1);
182  #define cmdIs(x,y) (0 == nextCmd.rfind(x, 0))
183  if(cmdIs("h", "help")){
184  parseHelp(nextCmd);
185  return;
186  }
187  if(!archSelected)setArch(nextCmd);
188  if(cmdIs("v", "verbose"))
189  parseVerbose(nextCmd);
190  else if(cmdIs("w", "warn"))
191  parseWarn(nextCmd);
192  else if(cmdIs("r", "region"))
193  parseRegion(nextCmd);
194  else if(cmdIs("b", "blank"))
195  parseBlank(nextCmd);
196  else if(cmdIs("c", "change"))
197  parseChange(nextCmd);
198  else if(cmdIs("d", "device"))
199  parseDevice(nextCmd);
200  else if(cmdIs("i", "input"))
201  parseInput(nextCmd);
202  else if(cmdIs("m", "merge"))
203  parseMerge(nextCmd);
204  else if(cmdIs("o", "output"))
205  parseOutput(nextCmd);
206  else if(cmdIs("a", "assembly"))
207  parseAssembly(nextCmd);
208  else if(cmdIs("e", "end"))
209  exit(0);
210  #undef cmdIs
211 }

232 int main(int argc, char * argv[]){
233  string command;
234  if(argc <= 1)
235  parseHelp("",1,1);//show usage and terminate with an error code
236  try {
237  bool readSTDIN = false;
238  for(int argi = 1 ; argi < argc ; argi++) {
239  if(argv[argi][0] == '-' && !command.empty()){ //new cmd and old cmd not empty
240  parseCommand(command);
241  command.clear();
242  }
243  if(command.empty())
244  command.append(argv[argi]);
245  else
246  command.append(" ").append(argv[argi]);
247  if(command == "-stdin"){
248  readSTDIN = true;
249  command.clear();
250  }
251  }
252  if(!command.empty())
253  parseCommand(command);
254  while(readSTDIN){
255  getline(cin,command);
256  if(command.at(command.find_first_not_of(" \t")) != '#'){//parse the command only if it doesnt start with '#'
257  if(string::npos != command.find("#"))
258  command = command.substr(0, command.find("#"));//if there is a comment in the current line, remove it
259  parseCommand(command);
260  }
261  }
262  } catch (const exception &e){
263  cout << "The program was terminated with message: \n\t'" << e.what() << "'\nWhile trying to execute command: \n\t'" << command << "'" << endl;
264  return 1;
265  }
266  return 0;
267 }
