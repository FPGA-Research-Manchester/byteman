58  command.clear();
59  }
60  }
61  if(!command.empty())
62  bytemanInstance.parseCommand(command);
63  while(readSTDIN){
64  getline(cin,command);
65  if(command.at(command.find_first_not_of(" \t")) != '#'){//parse the command only if it doesnt start with '#'
66  if(string::npos != command.find("#"))

70  }
71  } catch (const exception &e){
72  cout << "The program was terminated with message: \n\t'" << e.what() << "'\nWhile trying to execute command: \n\t'" << command << "'" << endl;

84 void byteman::parseCommand(string command)
85 {
86  command = str::replace(command, '=', ' ');
87  command = str::replace(command, ':', ' ');
88  if (command.at(0) == '-') command.erase(0, 1);
89  string params = str::findStringAndGetAllAfter(command, " ");
90  //cout <<"params is \""<<params<<"\""<<endl;
91  SelectedOptions options = parseParams(params);
92 
93  if(str::iff::firstStringWordIs(command, "h", "help"))//check if help first
94  byteman::help(params, 0);
95  else if(Architecture::Unknown == selectedArchitecture) //then check if architecture
96  byteman::setArchitecture(command);
97  else if(str::iff::firstStringWordIs(command, "v", "verbose")) //check the rest of commands
98  byteman::parseVerbose(params, options);
99  else if(str::iff::firstStringWordIs(command, "w", "warn"))
100  byteman::parseWarn(params, options);
101  else if(str::iff::firstStringWordIs(command, "r", "region"))
102  byteman::parseRegion(params, options);
103  else if(str::iff::firstStringWordIs(command, "b", "blank"))
104  byteman::parseBlank(params, options);
105  else if(str::iff::firstStringWordIs(command, "c", "change"))
106  byteman::parseChange(params, options);
107  else if(str::iff::firstStringWordIs(command, "d", "device"))
108  byteman::parseDevice(params, options);
109  else if(str::iff::firstStringWordIs(command, "i", "input"))
110  byteman::parseInput(params, options);
111  else if(str::iff::firstStringWordIs(command, "m", "merge"))
112  byteman::parseMerge(params, options);
113  else if(str::iff::firstStringWordIs(command, "o", "output"))
114  byteman::parseOutput(params, options);
115  else if(str::iff::firstStringWordIs(command, "a", "assembly"))
116  byteman::parseAssembly(params, options);
117  else if(str::iff::firstStringWordIs(command, "e", "end"))
118  exit(0);
119  else
120  throw runtime_error("Could not parse command. Consider checking out \"bytemap -help\".");
121 }

131 byteman::SelectedOptions byteman::parseParams(string params)
132 {
133  SelectedOptions options = SelectedOptions();
134  params = str::replace(params, ',', ' ');
135  stringstream ss(params);
136  string param;
137  while (!ss.eof()) {
138  ss >> param;
139  if("main" == param || "first" == param)options.mainBufferSelected++;
140  if("temp" == param || "second" == param)options.tempBufferSelected++;
141  param.clear();
142  }
143  if(0 == options.mainBufferSelected && 0 == options.tempBufferSelected) // by default, choose the main buffer
144  options.mainBufferSelected = 1;
145  return options;
146 }
