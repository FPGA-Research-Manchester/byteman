35 int main(int argc, char * argv[])
36 {
37  byteman bytemanInstance;
38  string command;
39  if(argc <= 1)
40  bytemanInstance.help("",1);//show usage and terminate with an error code
41  try {
42  bool readSTDIN = false;
43  for(int argi = 1 ; argi < argc ; argi++) {
44  if(argv[argi][0] == '-' && !command.empty()){ //new cmd and old cmd not empty
45  bytemanInstance.parse(command);
46  command.clear();
47  }
48  if(command.empty())
49  command.append(argv[argi]);
50  else
51  command.append(" ").append(argv[argi]);
52  if(command == "-stdin"){
53  readSTDIN = true;
54  command.clear();
55  }
56  }
57  if(!command.empty())
58  bytemanInstance.parse(command);
59  while(readSTDIN){
60  getline(cin,command);
61  if(command.at(command.find_first_not_of(" \t")) != '#'){//parse the command only if it doesnt start with '#'
62  if(string::npos != command.find("#"))
63  command = command.substr(0, command.find("#"));//if there is a comment in the current line, remove it
64  bytemanInstance.parse(command);
65  }
66  }
67  } catch (const exception &e){
68  cout << "The program was terminated with message: \n\t'" << e.what() << "'\nWhile trying to execute command: \n\t'" << command << "'" << endl;
69  return 1;
70  }
71  return 0;
72 }

89 void byteman::parse(string command)
90 {
91  command = str::replace(command, '=', ' ');
92  command = str::replace(command, ':', ' ');
93  if (command.at(0) == '-') command.erase(0, 1);
94  string params = str::findStringAndGetAllAfter(command, " ");
95  //cout <<"params is \""<<params<<"\""<<endl;
96  SelectedOptions options = parseParams(params);
97 
98  if(str::iff::firstStringWordIs(command, "h", "help"))//check if help first
99  byteman::help(params, 0);
100  else if(Architecture::Unknown == selectedArchitecture) //then check if architecture
101  byteman::setArchitecture(command);
102  else if(str::iff::firstStringWordIs(command, "v", "verbose")) //check the rest of commands
103  byteman::parseVerbose(params, options);
104  else if(str::iff::firstStringWordIs(command, "w", "warn"))
105  byteman::parseWarn(params, options);
106  else if(str::iff::firstStringWordIs(command, "r", "region"))
107  byteman::parseRegion(params, options);
108  else if(str::iff::firstStringWordIs(command, "b", "blank"))
109  byteman::parseBlank(params, options);
110  else if(str::iff::firstStringWordIs(command, "c", "change"))
111  byteman::parseChange(params, options);
112  else if(str::iff::firstStringWordIs(command, "d", "device"))
113  byteman::parseDevice(params, options);
114  else if(str::iff::firstStringWordIs(command, "i", "input"))
115  byteman::parseInput(params, options);
116  else if(str::iff::firstStringWordIs(command, "m", "merge"))
117  byteman::parseMerge(params, options);
118  else if(str::iff::firstStringWordIs(command, "o", "output"))
119  byteman::parseOutput(params, options);
120  else if(str::iff::firstStringWordIs(command, "a", "assembly"))
121  byteman::parseAssembly(params, options);
122  else if(str::iff::firstStringWordIs(command, "e", "end"))
123  exit(0);
124  else
125  throw runtime_error("Could not parse command. Consider checking out \"bytemap -help\".");
126 }

136 byteman::SelectedOptions byteman::parseParams(string params)
137 {
138  SelectedOptions options = SelectedOptions();
139  params = str::replace(params, ',', ' ');
140  stringstream ss(params);
141  string param;
142  while (!ss.eof()) {
143  ss >> param;
144  if("main" == param || "first" == param)options.mainBufferSelected++;
145  if("temp" == param || "second" == param)options.tempBufferSelected++;
146  param.clear();
147  }
148  if(0 == options.mainBufferSelected && 0 == options.tempBufferSelected) // by default, choose the main buffer
149  options.mainBufferSelected = 1;
150  return options;
151 }
