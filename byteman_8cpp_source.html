54  command.clear();
55  }
56  }
57  if(!command.empty())
58  bytemanInstance.parseCommand(command);
59  while(readSTDIN){
60  getline(cin,command);
61  if(command.at(command.find_first_not_of(" \t")) != '#'){//parse the command only if it doesnt start with '#'

67  } catch (const exception &e){

80 void byteman::parseCommand(string command)
81 {
82  command = StringFuncs::replace(command, '=', ' ');
83  command = StringFuncs::replace(command, ':', ' ');
84  if (command.at(0) == '-') command.erase(0, 1);
85  string params = StringFuncs::findStringAndGetAllAfter(command, " ");
86  //cout <<"params is \""<<params<<"\""<<endl;
87  SelectedOptions options = parseParams(params);
88 
89  if(StringFuncs::checkIf::firstStringWordIs(command, "h", "help"))//check if help first
90  byteman::help(params, 0);
91  else if(Architecture::Unknown == selectedArchitecture) //then check if architecture
92  byteman::setArchitecture(command);
93  else if(StringFuncs::checkIf::firstStringWordIs(command, "v", "verbose")) //check the rest of commands
94  byteman::parseVerbose(params, options);
95  else if(StringFuncs::checkIf::firstStringWordIs(command, "w", "warn"))
96  byteman::parseWarn(params, options);
97  else if(StringFuncs::checkIf::firstStringWordIs(command, "r", "region"))
98  byteman::parseRegion(params, options);
99  else if(StringFuncs::checkIf::firstStringWordIs(command, "b", "blank"))
100  byteman::parseBlank(params, options);
101  else if(StringFuncs::checkIf::firstStringWordIs(command, "c", "change"))
102  byteman::parseChange(params, options);
103  else if(StringFuncs::checkIf::firstStringWordIs(command, "d", "device"))
104  byteman::parseDevice(params, options);
105  else if(StringFuncs::checkIf::firstStringWordIs(command, "i", "input"))
106  byteman::parseInput(params, options);
107  else if(StringFuncs::checkIf::firstStringWordIs(command, "m", "merge"))
108  byteman::parseMerge(params, options);
109  else if(StringFuncs::checkIf::firstStringWordIs(command, "o", "output"))
110  byteman::parseOutput(params, options);
111  else if(StringFuncs::checkIf::firstStringWordIs(command, "a", "assembly"))
112  byteman::parseAssembly(params, options);
113  else if(StringFuncs::checkIf::firstStringWordIs(command, "e", "end"))
114  exit(0);
115  else
116  throw runtime_error("Could not parse command. Consider checking out \"bytemap -help\".");
117 }

127 byteman::SelectedOptions byteman::parseParams(string params)
128 {
129  SelectedOptions options = SelectedOptions();
130  params = StringFuncs::replace(params, ',', ' ');
131  stringstream ss(params);
132  string param;
133  while (!ss.eof()) {
134  ss >> param;
135  if("main" == param || "first" == param)options.mainBufferSelected++;
136  if("temp" == param || "second" == param)options.tempBufferSelected++;
137  param.clear();
138  }
139  if(0 == options.mainBufferSelected && 0 == options.tempBufferSelected) // by default, choose the main buffer
140  options.mainBufferSelected = 1;
141  return options;
142 }
