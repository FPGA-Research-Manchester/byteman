145 void XilinxSeries7::writeBitstream(string filename, string params, Rect2D cmdRect)
146 {
147  size_t dotpos = filename.rfind(".");
148  if(dotpos == string::npos)
149  throw runtime_error(string("Invalid file name: \"").append(filename).append("\"!\n"));
150  designName = filename.substr(0, dotpos);
151 
152  log("Writing Xilinx Series 7 bitstream to file \"" + filename + "\":");
153 
154  parseParams(params);
155 
156  if(selectedOptions.partialNotFull) {
157  if((cmdRect.position.row % XS7_CLB_PER_CLOCK_REGION != 0) || (cmdRect.size.row % XS7_CLB_PER_CLOCK_REGION != 0))
158  throw runtime_error("Currently only full clock region height relocations are supported (use row numbers multiple of 60).");
159  if(cmdRect.size.row <= 0 || cmdRect.size.col <= 0)
160  throw runtime_error("Invalid output size dimentions.");
161  }
162 
163  ofstream fout (filename, ofstream::binary | ofstream::trunc);
164  if(!fout.good())
165  throw runtime_error(string("Could not open file: \"").append(filename).append("\"!\n"));
166  if(str::iff::stringEndsWith(filename, ".bit"))
167  writeBitstreamBIT(fout, cmdRect);
168  else if(str::iff::stringEndsWith(filename, ".bin"))
169  writeBitstreamBIN(fout, cmdRect);
170  else
171  throw runtime_error(string("Unknown Xilinx Series 7 file format tried to be written.\n"));
172  fout.close();
173  log("Xilinx Series 7 bitstream file \"" + filename + "\" written successfully.");
174 }
