22  minorAddress++;
23  if((columnAddress == numberOfCols)?(minorAddress == XUSP_EXTRA_FRAMES_PER_ROW):(minorAddress == numberOfFramesPerResourceLetter[(uint8_t)resourceString[rowAddress][columnAddress]])){
