22 inline void XilinxUltraScalePlus::CAP_IncrementFAR_BlockType0(int slrID, int& blockType, int& rowAddress, int& columnAddress, int& minorAddress)
23 {
24  minorAddress++;
25  if((columnAddress == numberOfCols)?(minorAddress == XUSP_EXTRA_FRAMES_PER_ROW):(minorAddress == LUT_numberOfFramesForResourceLetter[(uint8_t)resourceString[rowAddress][columnAddress]])){
26  minorAddress = 0;
27  columnAddress++;
28  if(columnAddress == (numberOfCols+1)){
29  columnAddress = 0;
30  if(rowAddress == SLRinfo[slrID].toRow){
31  rowAddress = SLRinfo[slrID].fromRow;
32  blockType++;
33  } else {
34  rowAddress++;
35  }
36  }
37  }
38 }

92 inline void XilinxUltraScalePlus::CAP_parseFAR(int farValue, int slr, int& blockType, int& rowAddress, int& columnAddress, int& minorAddress)
93 {
94  blockType = (farValue >> 24) & 0x7;
95  rowAddress = (farValue >> 18) & 0x3F;
96  rowAddress += SLRinfo[slr].fromRow;
97  columnAddress = (farValue >> 8) & 0x3FF;
98  minorAddress = farValue & 0xFF;
99 }
101 inline uint32_t XilinxUltraScalePlus::CAP_makeFAR(int slr, int blockType, int rowAddress, int columnAddress, int minorAddress)
102 {
103  return (((blockType & 0x7) << 24) | (((rowAddress - SLRinfo[slr].fromRow) & 0x3F) << 18) | ((columnAddress & 0x3FF) << 8) | (minorAddress & 0xFF));
104 }
106 inline uint32_t XilinxUltraScalePlus::CAP_makeInstruction(int type, CAP::Operation operation, CAP::Register reg, int payload)
107 {
108  return (((type & 0x7) << 29) | ((static_cast<int>(operation) & 0x3) << 27) | ((static_cast<int>(reg) & 0x1F) << 13) | (payload & 0x07FFFFFF));
109 }
