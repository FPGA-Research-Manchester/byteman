37 void XilinxSeries7::assembler(string filenameIn, string filenameOut)
38 {
39  enum FILEFORMAT {FILE_NULL = 0, FILE_BIT, FILE_BIT_ASM};
40  FILEFORMAT fileformatIn = FILE_NULL, fileformatOut = FILE_NULL;
41 
42  if(str::iff::stringEndsWith(filenameIn, ".bit"))
43  fileformatIn = FILE_BIT;
44  if(str::iff::stringEndsWith(filenameIn, ".bitasm"))
45  fileformatIn = FILE_BIT_ASM;
46  if(str::iff::stringEndsWith(filenameOut, ".bit"))
47  fileformatOut = FILE_BIT;
48  if(str::iff::stringEndsWith(filenameOut, ".bitasm"))
49  fileformatOut = FILE_BIT_ASM;
50  if(fileformatIn == FILE_NULL)
51  throw runtime_error(string("Unknown Xilinx Series 7 file format tried to be read by the assembler. See \"byteman -h assembly\".\n"));
52  if(fileformatOut == FILE_NULL)
53  throw runtime_error(string("Unknown Xilinx Series 7 file format tried to be written by the assembler. See \"byteman -h assembly\".\n"));
54  if(fileformatIn == fileformatOut)
55  throw runtime_error(string("Unknown Xilinx Series 7 assembler operation between identical file formats. See \"byteman -h assembly\".\n"));
56 
57 
58  ifstream fin (filenameIn, ifstream::binary);
59  if(!fin.good())
60  throw runtime_error(string("Could not open file: \"").append(filenameIn).append("\" .\n"));
61 
62  ofstream fout (filenameOut, ofstream::binary | ofstream::trunc);
63  if(!fout.good())
64  throw runtime_error(string("Could not open file: \"").append(filenameOut).append("\"!\n"));
65 
66  if(fileformatIn == FILE_BIT && fileformatOut == FILE_BIT_ASM)
67  disassemblerBitToAsm(fin, fout);
68  if(fileformatIn == FILE_BIT_ASM && fileformatOut == FILE_BIT)
69  assemblerAsmToBit(fin, fout);
70 
71  fin.close();
72  fout.close();
73 }

86  if(str::iff::stringContains(line, "HEADER END"))break;
87  }
88  log("Design name: " + designName);
89  log("FPGA: " + partName);
90  log("Date: " + fileDate);

93 void XilinxSeries7::assemblerAsmToBit(ifstream& fin, ofstream& fout)
94 {
95  assemblerParseHeader(fin);
96 
97  setDeviceByPartNameOrThrow();
98  initializeResourceStringParameters();
99 
100  outputBITheader(fout, Endianness::BE);
101 
102  outputCAPheaderConstant(fout, Endianness::BE);
103 
104  XCAP::Register regAddr = XCAP::Register::UNDEFINED;
105  int frameCount = 0;
106  int slr = 0, b = 7, r = 0, c = 0, m = 0;
107  for (string line; getline(fin, line); ) {
108  if(line.at(line.find_first_not_of(" \t")) == '#')// if #, skip that line as comment
109  continue;
110  transform(line.begin(), line.end(),line.begin(), ::toupper);
111  replace(line.begin(), line.end(), '=', ' ');
112  replace(line.begin(), line.end(), '#', ' ');
113  replace(line.begin(), line.end(), ',', ' ');
114  if(str::iff::stringContains(line, "SYNC") && (!str::iff::stringContains(line, "DESYNC"))){
115  XCAP_writeSYNQ(fout, Endianness::BE);
116  }
117  else if(str::iff::stringContains(line, ".WORD")){
118  uint32_t wordValue;
119  if(!str::parse::multipleUints(line, wordValue))wordValue = 0;
120  FileIO::write32(fout, wordValue, Endianness::BE);
121  }
122  else if(str::iff::stringContains(line, "NOP")){
123  int nopHiddenValue;
124  if(!str::parse::multipleInts(line, nopHiddenValue))nopHiddenValue = 0;
125  XCAP_writeNOP(fout, 1, nopHiddenValue, Endianness::BE);
126  }
127  else if(str::iff::stringContains(line, "RESERVED")){
128  int reservedHiddenValue;
129  if(!str::parse::multipleInts(line, reservedHiddenValue))reservedHiddenValue = 0;
130  XCAP_writeRESERVED(fout, 1, reservedHiddenValue, Endianness::BE);
131  }
132  else if(str::iff::stringContains(line, "SELECT NEXT SLR")){
133  slr++;
134  int magic1size;
135  if(!str::parse::multipleInts(line, magic1size))magic1size = 0;
136  XCAP_writeSelectRegister(fout, XCAP::Register::MAGIC1, Endianness::BE);
137  XCAP_writeType2(fout, magic1size, Endianness::BE);
138  }
139  else if(str::iff::stringContains(line, "@")){
140  XCAP::Register newRegAddr = getXCAPregister(line);
141  if(str::iff::stringContains(line, "READ REG @")){
142  int readLength;
143  if(!str::parse::multipleInts(line, readLength)) readLength = 1; // default read length is 1 if unspecified
144  XCAP_writeReadRegister(fout, newRegAddr, readLength, Endianness::BE);
145  } else {// must be a write then ¯\_(ツ)_/¯
146 
147  if(newRegAddr == XCAP::Register::UNDEFINED){
148  throw runtime_error(string("Couldn't parse assembly command: \"").append(line).append("\"!"));
149  } else if(str::iff::stringContains(line, "SELECT REGISTER")){
150  XCAP_writeSelectRegister(fout, newRegAddr, Endianness::BE);
151  } else if(newRegAddr == XCAP::Register::FDRI){
152  if(!str::parse::multipleInts(line, frameCount))
153  throw runtime_error(string("FDRI command needs size: \"").append(line).append("\"!"));
154  int wordCount = frameCount * XS7_WORDS_PER_FRAME;
155  if(regAddr == XCAP::Register::FDRI) {
156  XCAP_writeType2(fout, wordCount, Endianness::BE);
157  } else {
158  XCAP_writeFDRI1(fout, wordCount, Endianness::BE);
159  }
160  if(frameCount > 0){
161  string frameLine;
162  for (string frameLine; getline(fin, frameLine); ) {
163  if(frameLine.at(frameLine.find_first_not_of(" \t")) == '#')// if #, skip that line as comment
164  continue;
165  uint32_t frameData[XS7_WORDS_PER_FRAME];
166  if(!str::parse::arrayOfUints(frameLine, XS7_WORDS_PER_FRAME, &frameData[0]))
167  throw runtime_error(string("Was expecting the data for a full frame on this line: \"").append(frameLine).append("\", because I think there are ").append(to_string(frameCount)).append(" frames left."));
168  //first 3 words are clk, next 90 are the 0-45 and 48-93 data words
169  for (int w = XS7_WORDS_AT_CLK; w < (XS7_WORDS_AT_CLK + XS7_WORDS_BEFORE_CLK) ; w++){
170  FileIO::write32(fout, frameData[w], Endianness::BE);
171  }
172  for (int w = 0; w < XS7_WORDS_AT_CLK ; w++){
173  FileIO::write32(fout, frameData[w], Endianness::BE);
174  }
175  for (int w = (XS7_WORDS_AT_CLK + XS7_WORDS_BEFORE_CLK); w < XS7_WORDS_PER_FRAME ; w++){
176  FileIO::write32(fout, frameData[w], Endianness::BE);
177  }
178 
179  frameCount--;
180  if(frameCount == 0)
181  break;
182  }
183  if(frameCount != 0)
184  throw runtime_error(string("End of file reached while performing FDRI frame writes.!"));
185  }
186  } else if(newRegAddr == XCAP::Register::FAR){
187  if(!str::parse::multipleInts(line, b, r, c, m))
188  throw runtime_error(string("Could not parse the new FAR value: \"").append(line).append("\"!"));
189  r += SLRinfo[slr].fromRow;
190  uint32_t farValue = XCAP_getFAR(slr, b, r, c, m);
191  XCAP_writeRegister(fout, XCAP::Register::FAR, farValue, Endianness::BE);
192  } else {
193  uint32_t newValue;
194  if(!str::parse::multipleUints(line, newValue))
195  throw runtime_error(string("Could not parse the new register value: \"").append(line).append("\"!"));
196  XCAP_writeRegister(fout, newRegAddr, newValue, Endianness::BE);
197  }
198  }
199  regAddr = newRegAddr;
200  } else {// ! str::iff::stringContains(line, "@")
201  //must have been a command then
202  XCAP::Command cmdID = getXCAPcommand(line);
203  if(cmdID == XCAP::Command::UNDEFINED)
204  throw runtime_error(string("Couldn't parse assembly command: \"").append(line).append("\"!"));
205  XCAP_writeCommand(fout, cmdID, Endianness::BE);
206  }
207  }
208  outputBITheaderLengthField(fout, Endianness::BE);
209 }

221 void XilinxSeries7::disassemblerBitToAsm(ifstream& fin, ofstream& fout)
222 {
223  loadedBitstreamEndianness = parseBitstreamEndianness(fin);
224  parseBITheader(fin, loadedBitstreamEndianness);
225  setDeviceByPartNameOrThrow();
226 
227  initializeResourceStringParameters();
228  disassemblerWriteHeader(fout);
229 
230  XCAP::Register regAddr = XCAP::Register::UNDEFINED;
231  bool synched = false;
232  int wordCount = 0;
233  int shadowFrameValid = 0;
234  int slr = 0, b = 7, r = 0, c = 0, m = 0;
235  //Parse bitstream
236  for( ; ; ){
237  uint32_t instruction = FileIO::read32(fin, Endianness::BE);
238  if(!fin.good()){
239  break; // done with the bitstream
240  } else {
241  if(! synched){
242  if(0xAA995566 == instruction){
243  synched = true;
244  fout << "SYNC" << endl;
245  } else {
246  fout << ".word " <<"0x" << uppercase << hex << setw(8) << setfill('0') << instruction << endl;
247  }
248  } else {
249  int instructionType = XCAP_getInstructionType(instruction);
250  XCAP::Operation instructionOPCODE = XCAP_getInstructionOperation(instruction);
251  int instructionPayload = XCAP_getInstructionPayload(instruction);
252  if(instructionType == 1) {
253  wordCount = XCAP_getInstructionWordCount(instruction);
254  regAddr = XCAP_getInstructionRegister(instruction);
255  } else if(instructionType == 2) {
256  wordCount = instructionPayload;
257  } else {
258  fout << "0x" << uppercase << hex << setw(8) << setfill('0') << instruction << " (Invalid instruction [invalid type])" << endl;
259  continue;
260  }
261 
262  if(instructionOPCODE == XCAP::Operation::NOP) {
263  if(instructionPayload != 0)
264  fout << "NOP #" << instructionPayload << endl;
265  else
266  fout << "NOP" << endl;
267  } else if(instructionOPCODE == XCAP::Operation::RESERVED) {
268  if(instructionPayload != 0)
269  fout << "RESERVED #" << instructionPayload << endl;
270  else
271  fout << "RESERVED" << endl;
272  } else if(instructionOPCODE == XCAP::Operation::READ) {
273  fout << "Read Reg @";
274  writeXCAPregisterName(fout, regAddr);
275  fout << " for length #" << wordCount << endl;
276  } else { // XCAP::Operation::WRITE
277  if((regAddr == XCAP::Register::FDRI) && (wordCount > 0) && (wordCount % XS7_WORDS_PER_FRAME == 0)) {
278  if(shadowFrameValid) {
279  fout << dec << "# Shadow register contents are written to frame (BlockType="<<b<<", GlobalRowAddress="<<r<<", MajorAddress="<<c<<", MinorAddress="<<m<<") (Frame type: " << getFrameType(b,r,c) << ")."<<endl;
280  XCAP_IncrementFAR(slr, b, r, c, m);
281  }
282  shadowFrameValid = 1;
283  int frameCount = (wordCount/XS7_WORDS_PER_FRAME);
284  fout << dec << "@FDRI for #" << frameCount << " frames:" << endl;
285  for(int i = 0 ; i < frameCount ; i++){
286  fout << "# ";
287  if(i == (frameCount-1)) fout << "(This frame data is written to shadow register!)";
288  fout << dec << "Writing frame #" << i << " (BlockType="<<b<<", GlobalRowAddress="<<r<<", MajorAddress="<<c<<", MinorAddress="<<m<<") (Frame type: " << getFrameType(b,r,c) << ") hex data:"<<endl;
289  uint32_t frameData[XS7_WORDS_PER_FRAME];
290  for(int w = 0 ; w < XS7_WORDS_PER_FRAME ; w++){
291  frameData[w] = FileIO::read32(fin, Endianness::BE);
292  }
293  fout<<"CLOCK: ";
294  for(int w = XS7_WORDS_BEFORE_CLK ; w < (XS7_WORDS_BEFORE_CLK + XS7_WORDS_AT_CLK) ; w++){
295  fout<< "0x" << uppercase << hex << setw(8) << setfill('0') << frameData[w] << " ";
296  }
297  fout<<" ;DATA: ";
298  for(int w = 0 ; w < XS7_WORDS_BEFORE_CLK ; w++){
299  fout<< "0x" << uppercase << hex << setw(8) << setfill('0') << frameData[w] << " ";
300  }
301  for(int w = (XS7_WORDS_BEFORE_CLK + XS7_WORDS_AT_CLK) ; w < XS7_WORDS_PER_FRAME ; w++){
302  fout<< "0x" << uppercase << hex << setw(8) << setfill('0') << frameData[w] << " ";
303  }
304  fout << endl;
305  XCAP_IncrementFAR(slr, b, r, c, m);
306  }
307  } else if(regAddr == XCAP::Register::CMD && wordCount == 1){
308  uint32_t writeData = FileIO::read32(fin, Endianness::BE);
309  writeXCAPcommandName(fout, static_cast<XCAP::Command>(writeData));
310  fout << ";";
311  if(XCAP::Command::DESYNC == static_cast<XCAP::Command>(writeData)){
312  shadowFrameValid = 0;
313  synched = false;
314  }
315  if(XCAP::Command::WCFG == static_cast<XCAP::Command>(writeData)){
316  shadowFrameValid = 0;
317  fout << " (also clears shadow register)";
318  }
319  fout << endl;
320  } else if(regAddr == XCAP::Register::MAGIC1){
321  uint32_t nextInstr = FileIO::read32(fin, Endianness::BE);
322  int nextInstrType = XCAP_getInstructionType(nextInstr);
323  XCAP::Operation nextInstrOP = XCAP_getInstructionOperation(nextInstr);
324  int nextInstrPayload = XCAP_getInstructionPayload(nextInstr);
325  if(2 == nextInstrType && XCAP::Operation::WRITE == nextInstrOP && 0 < nextInstrPayload){
326  slr++;
327  synched = false;
328  fout << "Select next SLR for the next #"<<dec<<nextInstrPayload<<" words." << endl;
329  } else {
330  fout << "Bad MAGIC1 instruction" << endl;
331  fin.seekg(-4,ios::cur);//rewind next instruction if not
332  }
333  } else if((instructionType == 1) && (wordCount == 0)){
334  fout << "Select register @";
335  writeXCAPregisterName(fout, regAddr);
336  fout << endl;
337  } else if(wordCount == 1){
338  uint32_t writeData = FileIO::read32(fin, Endianness::BE);
339  fout << "@";
340  writeXCAPregisterName(fout, regAddr);
341  if(regAddr == XCAP::Register::FAR) {
342  XCAP_parseFAR(writeData, slr, b, r, c, m);
343  fout << " = BlockType=" << dec <<b<<" RowAddress="<<(r-SLRinfo[slr].fromRow)<<" MajorAddress="<<c<<" MinorAddress="<<m<< endl;
344  } else {
345  fout << " = 0x" << uppercase << hex << setw(8) << setfill('0') << writeData << endl;
346  }
347  } else {
348  fout << "0x"<< uppercase << hex << setw(8) << setfill('0') << instruction << "(Bad instruction)" << endl;
349  }
350  } // OP_WRITE
351  }// if synched
352  } // if fin is still good
353  } // for(;;)
354 }
