49 void XilinxUltraScalePlus::ensureInitializedBitstreamArrays(){
50  if(partName == "")
51  throw runtime_error("The target device needs to be known, before you can manipulate any bitstream!");
52  if(initializedBitstreamPartName != partName){//The device is changed
53  initializedBitstreamPartName = partName;
54  if(bitstreamGlobal != nullptr)
55  delete bitstreamGlobal;
56 
57  for((numberOfCols = 0, numberOfFramesBeforeCol[0] = 0, numberOfBRAMCols = 0) ; (uint8_t)resourceString[0][numberOfCols] ; numberOfCols++){
58  numberOfFramesBeforeCol[numberOfCols+1] = numberOfFramesBeforeCol[numberOfCols] + numberOfFramesPerResourceLetter[(uint8_t)resourceString[0][numberOfCols]];
59  numberOfBRAMsBeforeCol[numberOfCols] = numberOfBRAMCols;
60  if(resourceString[0][numberOfCols] == 'A' || resourceString[0][numberOfCols] == 'B' || resourceString[0][numberOfCols] == 'C' || resourceString[0][numberOfCols] == '1') //ABC1 are "BRAM_L", "BRAM", "BRAM_R", "NULL BRAM"
61  numberOfBRAMCols++;
62  }
63  numberOfFramesPerRow = numberOfFramesBeforeCol[numberOfCols];
64  numberOfWordsPerRow = numberOfFramesPerRow * XUSP_WORDS_PER_FRAME;
65  //Allocate
66  bitstreamGlobal = new uint32_t[numberOfRows * (numberOfWordsPerRow + numberOfBRAMCols * XUSP_FRAMES_PER_BRAM_CONTENT_COLUMN * XUSP_WORDS_PER_FRAME) + 27*XUSP_WORDS_PER_FRAME];//TODO: WHY 27 ?! 27* extra frame for safety
67 
68  //Calc pointers
69  int offset = 0;
70  for(int r = 0 ; r < numberOfRows ; r++)
71  for(int c = 0 ; c < numberOfCols ; c++){
72  bitstreamCLB[r][c] = &bitstreamGlobal[offset];
73  offset += numberOfFramesPerResourceLetter[(uint8_t)resourceString[r][c]] * XUSP_WORDS_PER_FRAME;
74  }
75 
76  for(int r = 0 ; r < numberOfRows ; r++)
77  for(int c = 0 ; c < numberOfBRAMCols ; c++){
78  bitstreamBRAM[r][c] = &bitstreamGlobal[offset];
79  offset += XUSP_FRAMES_PER_BRAM_CONTENT_COLUMN*XUSP_WORDS_PER_FRAME;
80  }
81  bitstreamEnd = &bitstreamGlobal[offset];
82  }

123 void XilinxUltraScalePlus::readBitstreamBIT(string filename)
124 {
125  if(verbose)cout<<"Reading Xilinx UltraScale+ bitstream from file \""<<filename<<"\":\n";
126  //parse .BIT header:
127  ifstream fin (filename, ifstream::binary);
128  if(!fin.good())
129  throw runtime_error(string("Could not open file: \"").append(filename).append("\" .\n"));
130  //.bit header c0nstant:
131  fileReadAndDefineBE(headerConstTextLength, fin, 16);
132  fileReadAndDefineChars(headerConstText, fin, headerConstTextLength);
133  fileReadAndDefineBE(headerConstText2Length, fin, 16);
134  fileReadAndDefineChars(headerConstText2, fin, headerConstText2Length);
135  if(warn && (headerConstTextLength != 9 || strcmp(headerConstText,"\x0F\xF0\x0F\xF0\x0F\xF0\x0F\xF0\0") != 0 || headerConstText2Length != 1 || strcmp(headerConstText2,"a\0") != 0))
136  cout << "warn: .bit c0nstant header differs from expected.\n";
137  delete headerConstText; // we dont need that
138  delete headerConstText2; // we dont need that
139 
140  //.bit header vars:
141  fileReadAndDefineBE(headerDesignNameLength, fin, 16);
142  fileReadAndDefineChars(designNameChars, fin, headerDesignNameLength);
143 
144  if(verbose)cout<<"\tName of design: "<<designNameChars<<endl;
145  designName = string(designNameChars);
146  for(int headerDone = 0 ; !headerDone ; ){
147  fileReadAndDefineBE(key, fin, 8);
148  switch(key){
149  case 'b': {
150  fileReadAndDefineBE(headerAttrLength, fin, 16);
151  fileReadAndDefineChars(partNameChars, fin, headerAttrLength);
152  if(verbose)cout<<"\tFPGA part: "<<partNameChars<<endl;
153  partName = string(partNameChars);
154  break;
155  }
156  case 'c': {
157  fileReadAndDefineBE(headerAttrLength, fin, 16);
158  fileReadAndDefineChars(date, fin, headerAttrLength);
159  if(verbose)cout<<"\tdate: "<<date<<endl;
160  delete date; //not needed
161  break;
162  }
163  case 'd': {
164  fileReadAndDefineBE(headerAttrLength, fin, 16);
165  fileReadAndDefineChars(time, fin, headerAttrLength);
166  if(verbose)cout<<"\ttime: "<<time<<endl;
167  delete time; //not needed
168  break;
169  }
170  case 'e': {
171  headerDone = 1;//last attribute in a .bit header
172  int reportedRemainingFileLength;
173  fileReadBE(reportedRemainingFileLength, fin, 32);
174  int tmpPos = (int)fin.tellg();
175  fin.seekg (0, fin.end);
176  int fileSize = (int)fin.tellg();
177  fin.seekg (tmpPos, fin.beg);
178  if(warn && fileSize != (tmpPos + reportedRemainingFileLength))
179  cout << "warn: .bit header contained inaccurate file length field.\n";
180  break;
181  }
182  }
183  }
184  setDevice(getDeviceByNameOrThrow(partName));
185  ensureInitializedBitstreamArrays();//initialize bitstream arrays before writing
186 
187  //Follow some 0xFF's and the bus width detection c0nstant "\xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF\x44\x00\x22\x11\xBB\x00\x00\x00\xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF"
188  //However we don't care about reading those
189 
190  //Find sync
191  for(int syncDetectionDone = 0 ; !syncDetectionDone ; ){
192  fileReadAndDefineBE(word, fin, 32);
193  if(word == 0xAA995566)
194  syncDetectionDone++;
195  else
196  fin.seekg(-3,ios::cur);
197  }
198 
199  int regAddr = 0;
200  int wordCount = 0;
201  char shadowFrame[XUSP_WORDS_PER_FRAME*4];
202  int shadowFrameValid = 0;
203  int b = 0, r = 0, c = 0, m = 0;
204  //Parse bitstream
205  for( ; ; ){
206  fileReadAndDefineBE(instruction, fin, 32);
207  if(!fin.good()){
208  break; // done with the bitstream
209  } else {
210  if((instruction & 0xE0000000) == 0x20000000){ //Type 1 instruction
211  regAddr = (instruction & 0x07FFE000) >> 13;
212  wordCount = (instruction & 0x7FF);
213  } else if((instruction & 0xE0000000) == 0x40000000){ //Type 2 instruction
214  wordCount = (instruction & 0x07FFFFFF);
215  }
216  if((instruction & 0x18000000) == 0x00000000){ ;// NOP
217  } else if((instruction & 0x18000000) == 0x08000000){ ;// Read register
218  } else if((instruction & 0x18000000) == 0x18000000){ ;// Reserved
219  } else if((instruction & 0x18000000) == 0x10000000){ //Write register
220  for( ; wordCount > 0 ; ){
221  if(regAddr == XUSP_CAP_FAR){
222  fileReadAndDefineBE(farValue, fin, 32);
223  wordCount--;
224  b = (farValue & 0x07000000) >> 24; // block type
225  r = (farValue & 0x00FC0000) >> 18; // row address
226  c = (farValue & 0x0003FF00) >> 8; // column address
227  m = (farValue & 0x000000FF); // minor address
228  } else if(regAddr == XUSP_CAP_FDRI){
229  if(wordCount % 93 != 0)
230  throw runtime_error("FDRI write of a partial frame was detected, which is currently not supported.");
231  if(shadowFrameValid){
232  if(b == 0){ // CLB
233  //perf test 1
234  //memcpy((char*)&bitstreamCLB[r*numberOfWordsPerRow + (numberOfFramesBeforeCol[c] + m)*XUSP_WORDS_PER_FRAME] ,&shadowFrame, XUSP_WORDS_PER_FRAME*4);
235  //perf test 2
236  //memcpy((char*)&bitstreamCLB[r][(numberOfFramesBeforeCol[c] + m)*XUSP_WORDS_PER_FRAME] ,&shadowFrame, XUSP_WORDS_PER_FRAME*4);
237  //perf test 3
238  memcpy((char*)&bitstreamCLB[r][c][m*XUSP_WORDS_PER_FRAME] ,&shadowFrame, XUSP_WORDS_PER_FRAME*4);
239 
240  //increment FAR
241  m++;
242  if(m == numberOfFramesPerResourceLetter[(uint8_t)resourceString[r][c]]){
243  m = 0;
244  c++;
245  if(c == numberOfCols){
246  c = 0;
247  r++;
248  if(r == numberOfRows){
249  r = 0;
250  b++;
251  }
252  }
253  }
254  } else if(b == 1) { // bram;
255  memcpy((char*)&bitstreamBRAM[r][c][m*XUSP_WORDS_PER_FRAME] ,&shadowFrame, XUSP_WORDS_PER_FRAME*4);
256  //increment FAR
257  m++;
258  if(m == XUSP_FRAMES_PER_BRAM_CONTENT_COLUMN){
259  m = 0;
260  c++;
261  if(c == numberOfBRAMCols){
262  c = 0;
263  r++;
264  if(r == numberOfRows){
265  r = 0;
266  b++;
267  }
268  }
269  }
270  }
271  }
272  int forwardShadowReg = (wordCount/XUSP_WORDS_PER_FRAME) - 1;//skip shadow reg for certain number of frames
273  if(forwardShadowReg > 0){
274  if(b == 0){ // CLB
275  fin.read((char*)&bitstreamCLB[r][c][m*XUSP_WORDS_PER_FRAME], forwardShadowReg * XUSP_WORDS_PER_FRAME * 4);
276  } else if(b == 1){
277  fin.read((char*)&bitstreamBRAM[r][c][m*XUSP_WORDS_PER_FRAME], forwardShadowReg * XUSP_WORDS_PER_FRAME * 4);
278  }
279  for(int i = 0 ; i < forwardShadowReg ; i++){
280  if(b == 0){
281  m++;
282  if(m == numberOfFramesPerResourceLetter[(uint8_t)resourceString[r][c]]){
283  m = 0;
284  c++;
285  if(c == numberOfCols){
286  c = 0;
287  r++;
288  if(r == numberOfRows){
289  r = 0;
290  b++;
291  }
292  }
293  }
294  } else if(b == 1){
295  m++;
296  if(m == XUSP_FRAMES_PER_BRAM_CONTENT_COLUMN){
297  m = 0;
298  c++;
299  if(c == numberOfBRAMCols){
300  c = 0;
301  r++;
302  if(r == numberOfRows){
303  r = 0;
304  b++;
305  }
306  }
307  }
308  }
309  }
310  }
311  wordCount -= forwardShadowReg * XUSP_WORDS_PER_FRAME;
312  shadowFrameValid = 1;
313  fin.read((char*)&shadowFrame, XUSP_WORDS_PER_FRAME*4);
314  wordCount -= XUSP_WORDS_PER_FRAME;
315 
316  } else if(regAddr == XUSP_CAP_CMD){
317  fileReadAndDefineBE(command, fin, 32);
318  wordCount--;
319  if(command == XUSP_CAP_CMD_WCFG){
320  shadowFrameValid = 0;
321  }
322  } else {
323  fileReadAndDefineBE(scrap, fin, 32);
324  wordCount--;
325  }
326  }//Write register count for loop
327  }//Write register instruction
328  }// not end of file
329  }// for loop parsing the rest of the bitstream
330 }

332 void XilinxUltraScalePlus::merge(XilinxUltraScalePlus* srcBitstream, string params, int srcRow, int srcCol, int sizeRow, int sizeCol, int dstRow, int dstCol){
333  //Parse merge params
334  //This function uses a pointer to the second XUSP instance, but this pointer gets optimized as it is statically provided in the main function
335  int reloCLK = 0, reloCLB = 0, reloBRAM = 0;
336  if(params.find("logic") != string::npos || params.find("clb") != string::npos)reloCLB++;
337  if(params.find("clock") != string::npos || params.find("clk") != string::npos)reloCLK++;
338  if(params.find("blockram") != string::npos || params.find("bram") != string::npos)reloBRAM++;
339 
340  if(srcRow%XUSP_CLB_PER_CLOCK_REGION != 0)
341  throw runtime_error("Currently only full clock region height relocations are supported (use row numbers multiple of 60).");
342  if(sizeRow%XUSP_CLB_PER_CLOCK_REGION != 0)
343  throw runtime_error("Currently only full clock region height relocations are supported (use row numbers multiple of 60).");
344  if(dstRow%XUSP_CLB_PER_CLOCK_REGION != 0)
345  throw runtime_error("Currently only full clock region height relocations are supported (use row numbers multiple of 60).");
346 
347  int srcR = srcRow / XUSP_CLB_PER_CLOCK_REGION;
348  int sizeR = sizeRow / XUSP_CLB_PER_CLOCK_REGION;
349  int dstR = dstRow / XUSP_CLB_PER_CLOCK_REGION;
350 
351  for(int r = 0 ; r < sizeR ; r++){
352  for(int c = 0 ; c < sizeCol ; c++){
353  if(resourceString[srcR+r][srcCol+c] != resourceString[dstR+r][dstCol+c]){//if any two letters are different
354  if(numberOfFramesPerResourceLetter[(uint8_t)resourceString[srcR+r][srcCol+c]] != numberOfFramesPerResourceLetter[(uint8_t)resourceString[dstR+r][dstCol+c]]){
355  //also the number of frames is different?!

357  } else {

361  }
362  }
363  }
364  }
365 
366  for(int r = 0 ; r < sizeR ; r++){
367  for(int c = 0 ; c < sizeCol ; c++){//relocate dst[dstR+r][dstCol+c] <= src[srcR+r][srcCol+c]
368  for(int m = 0 ; m < numberOfFramesPerResourceLetter[(uint8_t)resourceString[dstR+r][dstCol+c]] ; m++){
369  if(reloCLB && reloCLK){
370  memcpy((char*)&bitstreamCLB[dstR+r][dstCol+c][m*XUSP_WORDS_PER_FRAME] ,(char*)&srcBitstream->bitstreamCLB[srcR+r][srcCol+c][m*XUSP_WORDS_PER_FRAME], XUSP_WORDS_PER_FRAME*4);
371  } else {
372  if(reloCLB){
373  memcpy((char*)&bitstreamCLB[dstR+r][dstCol+c][m*XUSP_WORDS_PER_FRAME] ,(char*)&srcBitstream->bitstreamCLB[srcR+r][srcCol+c][m*XUSP_WORDS_PER_FRAME], XUSP_WORDS_BEFORE_CLK*4);
374  memcpy((char*)&bitstreamCLB[dstR+r][dstCol+c][m*XUSP_WORDS_PER_FRAME+XUSP_WORDS_BEFORE_CLK+XUSP_WORDS_AT_CLK] ,(char*)&srcBitstream->bitstreamCLB[srcR+r][srcCol+c][m*XUSP_WORDS_PER_FRAME+XUSP_WORDS_BEFORE_CLK+XUSP_WORDS_AT_CLK], XUSP_WORDS_AFTER_CLK*4);

379  }//-if(reloCLK)

456 void XilinxUltraScalePlus::outputBITfooter(ofstream& fout){
457
