62 XilinxUltraScalePlus::SelectedOptions XilinxUltraScalePlus::parseParams(string params){
63  SelectedOptions options = SelectedOptions();
64  replace(params.begin(), params.end(), ',', ' ');
65  stringstream ss(params);
66  string param;
67  while (!ss.eof()) {
68  ss >> param;
69  if(param == "clock" || param == "clk")options.clk++;
70  if(param == "logic" || param == "clb")options.clb++;
71  if(param == "blockram" || param == "bram")options.bram++;
72  if(param == "blank")options.blank++;
73  if(param == "full")options.partial = 0;
74  if(param == "partial")options.partial = 1;
75  if(param == "set")options.op = MergeOP::SET;
76  if(param == "xor")options.op = MergeOP::XOR;
77  if(param == "or")options.op = MergeOP::OR;
78  if(param == "and")options.op = MergeOP::AND;
79  param.clear();
80  }
81  if(options.clk == 0 && options.clb == 0 && options.bram == 0){ // by default, choose all
82  options.clk = options.bram = options.clb = 1;
83  }
84  return options;
85 }

87 void XilinxUltraScalePlus::initializeResourceStringParameters(){
88  if(partName == "")
89  throw runtime_error("The target device needs to be known, before you can manipulate any bitstream!");
90  if(initializedResourceParamsPartName != partName){//The device is changed
91  initializedResourceParamsPartName = partName;
92 
93  for((numberOfCols = 0, numberOfFramesBeforeCol[0] = 0, numberOfBRAMCols = 0) ; (uint8_t)resourceString[0][numberOfCols] ; numberOfCols++){
94  numberOfFramesBeforeCol[numberOfCols+1] = numberOfFramesBeforeCol[numberOfCols] + numberOfFramesPerResourceLetter[(uint8_t)resourceString[0][numberOfCols]];
95  numberOfBRAMsBeforeCol[numberOfCols] = numberOfBRAMCols;
96  if(resourceString[0][numberOfCols] == 'A' || resourceString[0][numberOfCols] == 'B' || resourceString[0][numberOfCols] == 'C' || resourceString[0][numberOfCols] == '1') //ABC1 are "BRAM_L", "BRAM", "BRAM_R", "NULL BRAM"
97  numberOfBRAMCols++;
98  }
99  numberOfFramesPerRow = numberOfFramesBeforeCol[numberOfCols] + XUSP_EXTRA_FRAMES_PER_ROW;
100  numberOfWordsPerRow = numberOfFramesPerRow * XUSP_WORDS_PER_FRAME;
101  }

136 void XilinxUltraScalePlus::blank(string params){
137  XilinxUltraScalePlus::ensureInitializedBitstreamArrays();
138  int blankCLB = 0, blankBRAM = 0;
139  if(params.find("logic") != string::npos || params.find("clb") != string::npos)blankCLB++;
140  if(params.find("blockram") != string::npos || params.find("bram") != string::npos)blankBRAM++;
141  if(regionSelection.empty()) {
142  if(blankCLB)
143  memset(&bitstreamCLB[0][0][0], 0, (&bitstreamBRAM[0][0][0]-&bitstreamCLB[0][0][0]));
144  if(blankBRAM)
145  memset(&bitstreamBRAM[0][0][0], 0, (bitstreamEnd-&bitstreamBRAM[0][0][0]));
146  } else {
147  if(blankCLB)
148  for (Rect2D &rect : regionSelection) {
149  for(int r = rect.position.row ; r < (rect.position.row + rect.size.row) ; r++){
150  int blankSize = numberOfFramesBeforeCol[rect.position.col + rect.size.col] - numberOfFramesBeforeCol[rect.position.col];
151  blankSize *= XUSP_WORDS_PER_FRAME * 4;
152  memset(&bitstreamCLB[r][rect.position.col][0], 0, blankSize); //memset with size 0 is safe, no need to check anything
153  }
154  }
155  if(blankBRAM)
156  for (Rect2D &rect : regionSelection) {
157  for(int r = rect.position.row ; r < (rect.position.row + rect.size.row) ; r++){
158  int blankSize = numberOfBRAMsBeforeCol[rect.position.col + rect.size.col] - numberOfBRAMsBeforeCol[rect.position.col];
159  blankSize *= XUSP_FRAMES_PER_BRAM_CONTENT_COLUMN * XUSP_WORDS_PER_FRAME * 4;
160  memset(&bitstreamCLB[r][numberOfBRAMsBeforeCol[rect.position.col]][0], 0, blankSize); //memset with size 0 is safe, no need to check anything
161  }
162  }
163  }

165 void XilinxUltraScalePlus::ensureRowCompatibility(Coord2D src, int offsetRow, int sizeCol, Coord2D dst){
166  int srcRA = (src.row+offsetRow) / XUSP_CLB_PER_CLOCK_REGION;
167  int dstRA = (dst.row+offsetRow) / XUSP_CLB_PER_CLOCK_REGION;
168  for(int c = 0 ; c < sizeCol ; c++){
169  if(resourceString[srcRA][src.col+c] != resourceString[dstRA][dst.col+c]){//if any two letters are different
170  if(numberOfFramesPerResourceLetter[(uint8_t)resourceString[srcRA][src.col+c]] != numberOfFramesPerResourceLetter[(uint8_t)resourceString[dstRA][dst.col+c]]){
171  //also the number of frames is different?!
172  throw runtime_error(string("Tried to relocate to an incompatible location(").append(typeOfFramesPerResourceLetter[(uint8_t)resourceString[srcRA][src.col+c]]).append(" to frame type ").append(typeOfFramesPerResourceLetter[(uint8_t)resourceString[dstRA][dst.col+c]]).append(")."));
173  } else {
174  //okay, different col, but same width. Throw warning?
175  warn("Relocating from frame type " + string(typeOfFramesPerResourceLetter[(uint8_t)resourceString[srcRA][src.col+c]]) + " to frame type " + string(typeOfFramesPerResourceLetter[(uint8_t)resourceString[dstRA][dst.col+c]]) + ".");
176  }
177  }
178  }

180 void XilinxUltraScalePlus::ensureRegionCompatibility(Rect2D src, Coord2D dst){
181  for(int r = 0 ; r < src.size.row ; ){ //this loop will iterate south border and all (srcRow:srcRow+sizeRow) at CLBs 0 and 59
182  XilinxUltraScalePlus::ensureRowCompatibility(src.position, r, src.size.col, dst);
183  if( ((src.position.row + r) % XUSP_CLB_PER_CLOCK_REGION) == (XUSP_CLB_PER_CLOCK_REGION-1) ) { // isLastCLBinRow
184  r++;
185  } else if( ((src.position.row + r) % XUSP_CLB_PER_CLOCK_REGION) == 0 ) { // isFirstCLBinRow
186  r += (XUSP_CLB_PER_CLOCK_REGION - 2);
187  } else { // get to the next {isLastCLBinRow}
188  r += (XUSP_CLB_PER_CLOCK_REGION - 1 - ((src.position.row + r) % XUSP_CLB_PER_CLOCK_REGION));
189  }
190  }
191  ensureRowCompatibility(src.position, src.size.row-1, src.size.col, dst); //iterate north border as well

193 void XilinxUltraScalePlus::merge(XilinxUltraScalePlus* srcBitstream, string params, Rect2D src, Coord2D dst){
194  //This function uses a pointer to the second XUSP instance, but this pointer should get optimized as it is statically provided in the main function
195  SelectedOptions options = XilinxUltraScalePlus::parseParams(params);
196 
197  XilinxUltraScalePlus::ensureRegionCompatibility(src, dst);
198  if(src.position.row%XUSP_CLB_PER_CLOCK_REGION != 0)
199  throw runtime_error("Currently only full clock region height relocations are supported (use row numbers multiple of 60).");
200  if(src.size.row%XUSP_CLB_PER_CLOCK_REGION != 0)
201  throw runtime_error("Currently only full clock region height relocations are supported (use row numbers multiple of 60).");
202  if(dst.row%XUSP_CLB_PER_CLOCK_REGION != 0)
203  throw runtime_error("Currently only full clock region height relocations are supported (use row numbers multiple of 60).");
204 
205  Endianess endianDifference = Endian::diff(loadedBitstreamEndianess, srcBitstream->loadedBitstreamEndianess);
206 
207  if((Endianess::NATIVE == endianDifference) && MergeOP::SET == options.op)
208  fastMerge(srcBitstream, options, src, dst);
209  else
210  flexiMerge(srcBitstream, options, endianDifference, src, dst);
211 }

217 void XilinxUltraScalePlus::fastMerge(XilinxUltraScalePlus* srcBitstream, SelectedOptions options, Rect2D src, Coord2D dst){
218  int srcRA = src.position.row / XUSP_CLB_PER_CLOCK_REGION;
219  int sizeR = src.size.row / XUSP_CLB_PER_CLOCK_REGION;
220  int dstRA = dst.row / XUSP_CLB_PER_CLOCK_REGION;
221 
222  for(int r = 0 ; r < sizeR ; r++){
223  for(int c = 0 ; c < src.size.col ; c++){//relocate dst[dstRA+r][dst.col+c] <= src[srcRA+r][src.position.col+c]
224  for(int m = 0 ; m < numberOfFramesPerResourceLetter[(uint8_t)resourceString[dstRA+r][dst.col+c]] ; m++){
225  if(options.clb && options.clk){
226  memcpy((char*)&bitstreamCLB[dstRA+r][dst.col+c][m*XUSP_WORDS_PER_FRAME] ,(char*)&srcBitstream->bitstreamCLB[srcRA+r][src.position.col+c][m*XUSP_WORDS_PER_FRAME], XUSP_WORDS_PER_FRAME*4);
227  } else {
228  if(options.clb){
229  memcpy((char*)&bitstreamCLB[dstRA+r][dst.col+c][m*XUSP_WORDS_PER_FRAME] , (char*)&srcBitstream->bitstreamCLB[srcRA+r][src.position.col+c][m*XUSP_WORDS_PER_FRAME], XUSP_WORDS_BEFORE_CLK*4);
230  memcpy((char*)&bitstreamCLB[dstRA+r][dst.col+c][m*XUSP_WORDS_PER_FRAME+XUSP_WORDS_BEFORE_CLK+XUSP_WORDS_AT_CLK] ,(char*)&srcBitstream->bitstreamCLB[srcRA+r][src.position.col+c][m*XUSP_WORDS_PER_FRAME+XUSP_WORDS_BEFORE_CLK+XUSP_WORDS_AT_CLK], XUSP_WORDS_AFTER_CLK*4);
231  }//-if(reloCLB)
232  if(options.clk){
233  for(int w = XUSP_WORDS_BEFORE_CLK ; w < (XUSP_WORDS_BEFORE_CLK + XUSP_WORDS_AT_CLK) ; w++)
234  bitstreamCLB[dstRA+r][dst.col+c][m*XUSP_WORDS_PER_FRAME + w] = srcBitstream->bitstreamCLB[srcRA+r][src.position.col+c][m*XUSP_WORDS_PER_FRAME + w];
235  }//-if(reloCLK)
236  }//-if(reloCLB && reloCLK)
237  }
238  }
239  }
240  if(options.bram){
241  for(int r = 0 ; r < sizeR ; r++){
242  int bramCols = numberOfBRAMsBeforeCol[src.position.col + src.size.col] - numberOfBRAMsBeforeCol[src.position.col];
243  int srcCol = numberOfBRAMsBeforeCol[src.position.col];
244  int dstCol = numberOfBRAMsBeforeCol[dst.col];
245  memcpy((char*)&bitstreamBRAM[dstRA+r][dstCol][0] ,(char*)&srcBitstream->bitstreamBRAM[srcRA+r][srcCol], bramCols * XUSP_FRAMES_PER_BRAM_CONTENT_COLUMN * XUSP_WORDS_PER_FRAME * 4);
246  }
247  }//-if(reloBRAM)

249 void XilinxUltraScalePlus::flexiMerge(XilinxUltraScalePlus* srcBitstream, SelectedOptions options, Endianess endianConversionNeeded, Rect2D src, Coord2D dst){
250  int srcRA = src.position.row / XUSP_CLB_PER_CLOCK_REGION;
251  int sizeR = src.size.row / XUSP_CLB_PER_CLOCK_REGION;
252  int dstRA = dst.row / XUSP_CLB_PER_CLOCK_REGION;
253 
254  for(int r = 0 ; r < sizeR ; r++){
255  for(int c = 0 ; c < src.size.col ; c++){//relocate dst[dstRA+r][dst.col+c] <= src[srcRA+r][src.position.col+c]
256  for(int m = 0 ; m < numberOfFramesPerResourceLetter[(uint8_t)resourceString[dstRA+r][dst.col+c]] ; m++){
257  for(int w = 0 ; w < XUSP_WORDS_PER_FRAME ; w++){
258  if(XUSP_WORDS_BEFORE_CLK <= w && (XUSP_WORDS_BEFORE_CLK + XUSP_WORDS_AT_CLK) > w){
259  if(!options.clk)
260  continue;
261  } else {
262  if(!options.clb)
263  continue;
264  }
265  switch(options.op){
266  case MergeOP::SET:
267  bitstreamCLB[dstRA+r][dst.col+c][m*XUSP_WORDS_PER_FRAME + w] = Endian::NativeToAnyEndianess32(srcBitstream->bitstreamCLB[srcRA+r][src.position.col+c][m*XUSP_WORDS_PER_FRAME + w], endianConversionNeeded);
268  break;
269  case MergeOP::XOR:
270  bitstreamCLB[dstRA+r][dst.col+c][m*XUSP_WORDS_PER_FRAME + w] ^= Endian::NativeToAnyEndianess32(srcBitstream->bitstreamCLB[srcRA+r][src.position.col+c][m*XUSP_WORDS_PER_FRAME + w], endianConversionNeeded);
271  break;
272  case MergeOP::OR:
273  bitstreamCLB[dstRA+r][dst.col+c][m*XUSP_WORDS_PER_FRAME + w] |= Endian::NativeToAnyEndianess32(srcBitstream->bitstreamCLB[srcRA+r][src.position.col+c][m*XUSP_WORDS_PER_FRAME + w], endianConversionNeeded);
274  break;
275  case MergeOP::AND:
276  bitstreamCLB[dstRA+r][dst.col+c][m*XUSP_WORDS_PER_FRAME + w] &= Endian::NativeToAnyEndianess32(srcBitstream->bitstreamCLB[srcRA+r][src.position.col+c][m*XUSP_WORDS_PER_FRAME + w], endianConversionNeeded);
277  break;
278  }
279  }
280  }
281  }
282  }
283  if(options.bram){
284  for(int r = 0 ; r < sizeR ; r++){
285  int bramCols = numberOfBRAMsBeforeCol[src.position.col + src.size.col] - numberOfBRAMsBeforeCol[src.position.col];
286  int srcCol = numberOfBRAMsBeforeCol[src.position.col];
287  int dstCol = numberOfBRAMsBeforeCol[dst.col];
288  for(int c = 0 ; c < bramCols ; c++){
289  for(int w = 0 ; w < XUSP_WORDS_PER_FRAME * XUSP_FRAMES_PER_BRAM_CONTENT_COLUMN ; w++){
290  switch(options.op){
291  case MergeOP::SET:
292  bitstreamBRAM[dstRA+r][dstCol+c][w] = Endian::NativeToAnyEndianess32(srcBitstream->bitstreamCLB[srcRA+r][srcCol+c][w], endianConversionNeeded);
293  break;
294  case MergeOP::XOR:
295  bitstreamBRAM[dstRA+r][dstCol+c][w] ^= Endian::NativeToAnyEndianess32(srcBitstream->bitstreamCLB[srcRA+r][srcCol+c][w], endianConversionNeeded);
296  break;
297  case MergeOP::OR:
298  bitstreamBRAM[dstRA+r][dstCol+c][w] |= Endian::NativeToAnyEndianess32(srcBitstream->bitstreamCLB[srcRA+r][srcCol+c][w], endianConversionNeeded);
299  break;
300  case MergeOP::AND:
301  bitstreamBRAM[dstRA+r][dstCol+c][w] &= Endian::NativeToAnyEndianess32(srcBitstream->bitstreamCLB[srcRA+r][srcCol+c][w], endianConversionNeeded);
302  break;
303  }
304  }
305  }
306  }
307  }//-if(reloBRAM)
