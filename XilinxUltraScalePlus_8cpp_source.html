46 void XilinxUltraScalePlus::initializeResourceStringParameters(){
47  if(partName == "")
48  throw runtime_error("The target device needs to be known, before you can manipulate any bitstream!");
49  if(initializedResourceParamsPartName != partName){//The device is changed
50  initializedResourceParamsPartName = partName;
51 
52  for((numberOfCols = 0, numberOfFramesBeforeCol[0] = 0, numberOfBRAMCols = 0) ; (uint8_t)resourceString[0][numberOfCols] ; numberOfCols++){
53  numberOfFramesBeforeCol[numberOfCols+1] = numberOfFramesBeforeCol[numberOfCols] + numberOfFramesPerResourceLetter[(uint8_t)resourceString[0][numberOfCols]];
54  numberOfBRAMsBeforeCol[numberOfCols] = numberOfBRAMCols;
55  if(resourceString[0][numberOfCols] == 'A' || resourceString[0][numberOfCols] == 'B' || resourceString[0][numberOfCols] == 'C' || resourceString[0][numberOfCols] == '1') //ABC1 are "BRAM_L", "BRAM", "BRAM_R", "NULL BRAM"
56  numberOfBRAMCols++;
57  }
58  numberOfFramesPerRow = numberOfFramesBeforeCol[numberOfCols] + XUSP_EXTRA_FRAMES_PER_ROW;
59  numberOfWordsPerRow = numberOfFramesPerRow * XUSP_WORDS_PER_FRAME;
60  }
61 }

127 void XilinxUltraScalePlus::ensureRowCompatibility(int srcRow, int srcCol, int offsetRow, int sizeCol, int dstRow, int dstCol){
128  int srcRA = (srcRow+offsetRow) / XUSP_CLB_PER_CLOCK_REGION;
129  int dstRA = (dstRow+offsetRow) / XUSP_CLB_PER_CLOCK_REGION;
130  for(int c = 0 ; c < sizeCol ; c++){
131  if(resourceString[srcRA][srcCol+c] != resourceString[dstRA][dstCol+c]){//if any two letters are different
132  if(numberOfFramesPerResourceLetter[(uint8_t)resourceString[srcRA][srcCol+c]] != numberOfFramesPerResourceLetter[(uint8_t)resourceString[dstRA][dstCol+c]]){
133  //also the number of frames is different?!
134  throw runtime_error(string("Tried to relocate to an incompatible location(").append(typeOfFramesPerResourceLetter[(uint8_t)resourceString[srcRA][srcCol+c]]).append(" to frame type ").append(typeOfFramesPerResourceLetter[(uint8_t)resourceString[dstRA][dstCol+c]]).append(")."));
135  } else {
136  //okay, different col, but same width. Throw warning?
137  if(warn)
138  cout<<"Relocating from frame type "<<typeOfFramesPerResourceLetter[(uint8_t)resourceString[srcRA][srcCol+c]]<<" to frame type "<<typeOfFramesPerResourceLetter[(uint8_t)resourceString[dstRA][dstCol+c]]<<".\n";
139  }

143 void XilinxUltraScalePlus::ensureRegionCompatibility(int srcRow, int srcCol, int sizeRow, int sizeCol, int dstRow, int dstCol){
144  for(int r = 0 ; r < sizeRow ; ){ //this loop will iterate south border and all (srcRow:srcRow+sizeRow) at CLBs 0 and 59
145  XilinxUltraScalePlus::ensureRowCompatibility(srcRow, srcCol, r, sizeCol, dstRow, dstCol);
146  if( ((srcRow + r) % XUSP_CLB_PER_CLOCK_REGION) == (XUSP_CLB_PER_CLOCK_REGION-1) ) { // isLastCLBinRow
147  r++;
148  } else if( ((srcRow + r) % XUSP_CLB_PER_CLOCK_REGION) == 0 ) { // isFirstCLBinRow
149  r += (XUSP_CLB_PER_CLOCK_REGION - 2);
150  } else { // get to the next {isLastCLBinRow}
151  r += (XUSP_CLB_PER_CLOCK_REGION - 1 - ((srcRow + r) % XUSP_CLB_PER_CLOCK_REGION));
152  }
153  }
154  ensureRowCompatibility(srcRow, srcCol, sizeRow-1, sizeCol, dstRow, dstCol); //iterate north border as well
