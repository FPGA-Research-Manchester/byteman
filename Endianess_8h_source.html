63  inline Endianess swapEndian(Endianess e){
64  if(Endianess::BE == e)
65  return Endianess::LE;
66  if(Endianess::LE == e)
67  return Endianess::BE;
68  if(Endianess::BE_BS == e)
69  return Endianess::LE_BS;
70  if(Endianess::LE_BS == e)
71  return Endianess::BE_BS;
72  //Native then
73  if(IS_LITTLE_ENDIAN())
74  return Endianess::BE;
75  return Endianess::LE;
76  }

91  inline Endianess fixEndianIfNative(Endianess e){

105  inline Endianess diff(Endianess e1, Endianess e2){
106  Endianess accumulatedEndianessDiff = Endianess::NATIVE;
107 
108  if(Endianess::BE_BS == e1 || Endianess::LE_BS == e1)
109  accumulatedEndianessDiff = swapEndianBits(accumulatedEndianessDiff);
110  if(Endianess::BE_BS == e2 || Endianess::LE_BS == e2)
111  accumulatedEndianessDiff = swapEndianBits(accumulatedEndianessDiff);
112 
113  if(IS_LITTLE_ENDIAN()){ // If native system is little endian
114  if(Endianess::BE == e1 || Endianess::BE_BS == e1)
115  accumulatedEndianessDiff = swapEndian(accumulatedEndianessDiff);
116  if(Endianess::BE == e2 || Endianess::BE_BS == e2)
117  accumulatedEndianessDiff = swapEndian(accumulatedEndianessDiff);
118  } else { // If native system is big endian
119  if(Endianess::LE == e1 || Endianess::LE_BS == e1)
120  accumulatedEndianessDiff = swapEndian(accumulatedEndianessDiff);
121  if(Endianess::LE == e2 || Endianess::LE_BS == e2)
122  accumulatedEndianessDiff = swapEndian(accumulatedEndianessDiff);
123  }
124 
125  return fixEndianIfNative(accumulatedEndianessDiff);
126  }
