27 void XilinxUltraScalePlus::readBitstream(string filename)
28 {
29  ifstream fin (filename, ifstream::binary);
30  if(!fin.good())
31  throw runtime_error(string("Could not open file: \"").append(filename).append("\" .\n"));
32  if(verbose)cout<<"Reading Xilinx UltraScale+ bitstream from file \""<<filename<<"\":\n";
33  if(fileExtensionIs(filename, "bit"))
34  XilinxUltraScalePlus::readBitstreamBIT(fin);
35  else
36  throw runtime_error(string("Unknown Xilinx UltraScale+ file format tried to be read.\n"));
37  fin.close();

39 void XilinxUltraScalePlus::readBitstreamBIT(ifstream& fin)
40 {
41  XilinxUltraScalePlus::readBITheader(fin);
42  XilinxUltraScalePlus::setDevice(XilinxUltraScalePlus::getDeviceByNameOrThrow(partName), partName);
43  XilinxUltraScalePlus::ensureInitializedBitstreamArrays();//initialize bitstream arrays before writing
44 
45  int regAddr = 0;
46  int wordCount = 0;
47  char shadowFrame[XUSP_WORDS_PER_FRAME*4];
48  int shadowFrameValid = 0;
49  int b = 7, r = 0, c = 0, m = 0;
50  //Parse bitstream
51  for( ; ; ){
52  fileReadAndDefineBE(instruction, fin, 32);
53  if(!fin.good()){
54  break; // done with the bitstream
55  } else {
56  int instructionType = getInstructionType(instruction);
57  int instructionOPCODE = getInstructionOPCODE(instruction);
58  int instructionPayload = getInstructionPayload(instruction);
59  if(instructionType == 1) {
60  wordCount = getInstructionType1WordCount(instruction);
61  regAddr = getInstructionType1Register(instruction);
62  } else if(instructionType == 2) {
63  wordCount = instructionPayload;
64  } else {
65  throw runtime_error("Bitstream has invalid instruction (invalid type).");
66  }
67  if(instructionOPCODE == OP_WRITE){ //Write register
68  for( ; wordCount > 0 ; ){
69  if(regAddr == XUSP_CAP_REG_FAR){
70  fileReadAndDefineBE(farValue, fin, 32);
71  wordCount--;
72  parseFAR(farValue,b,r,c,m);
73  } else if(regAddr == XUSP_CAP_REG_FDRI){
74  if(wordCount % 93 != 0)
75  throw runtime_error("FDRI write of a partial frame was detected, which is currently not supported.");
76  if(shadowFrameValid){
77  if(b == 0){ // CLB
78  memcpy((char*)&bitstreamCLB[r][c][m*XUSP_WORDS_PER_FRAME] ,&shadowFrame, XUSP_WORDS_PER_FRAME*4);
79  } else if(b == 1) { // bram;
80  memcpy((char*)&bitstreamBRAM[r][c][m*XUSP_WORDS_PER_FRAME] ,&shadowFrame, XUSP_WORDS_PER_FRAME*4);
81  }
82  XUSP_INCR_FAR(b,r,c,m);
83  }
84  int forwardShadowReg = (wordCount/XUSP_WORDS_PER_FRAME) - 1;//skip shadow reg for certain number of frames
85  if(forwardShadowReg > 0){
86  if(b == 0){ // CLB
87  fin.read((char*)&bitstreamCLB[r][c][m*XUSP_WORDS_PER_FRAME], forwardShadowReg * XUSP_WORDS_PER_FRAME * 4);
88  } else if(b == 1){
89  fin.read((char*)&bitstreamBRAM[r][c][m*XUSP_WORDS_PER_FRAME], forwardShadowReg * XUSP_WORDS_PER_FRAME * 4);
90  }
91  for(int i = 0 ; i < forwardShadowReg ; i++){
92  XUSP_INCR_FAR(b,r,c,m);
93  }
94  }
95  wordCount -= forwardShadowReg * XUSP_WORDS_PER_FRAME;
96  shadowFrameValid = 1;
97  fin.read((char*)&shadowFrame, XUSP_WORDS_PER_FRAME*4);
98  wordCount -= XUSP_WORDS_PER_FRAME;
99 
100  } else if(regAddr == XUSP_CAP_REG_CMD){
101  fileReadAndDefineBE(command, fin, 32);
102  wordCount--;
103  if(command == XUSP_CAP_CMD_WCFG){
104  shadowFrameValid = 0;
105  }
106  } else {
107  fileReadAndDefineBE(scrap, fin, 32);
108  wordCount--;
109  }
110  }//Write register count for loop
111  }//Write register instruction
112  }// not end of file
113  }// for loop parsing the rest of the bitstream
114 }

122 void XilinxUltraScalePlus::readBITheader(ifstream& fin){
123  //.bit header c0nstant:
124  fileReadAndDefineBE(headerConstTextLength, fin, 16);
125  fileReadAndDefineChars(headerConstText, fin, headerConstTextLength);
126  fileReadAndDefineBE(headerConstText2Length, fin, 16);
127  fileReadAndDefineChars(headerConstText2, fin, headerConstText2Length);
128  if(warn && (headerConstTextLength != 9 || strcmp(headerConstText,"\x0F\xF0\x0F\xF0\x0F\xF0\x0F\xF0\0") != 0 || headerConstText2Length != 1 || strcmp(headerConstText2,"a\0") != 0))
129  cout << "warn: .bit c0nstant header differs from expected.\n";
130  delete headerConstText; // we dont need that
131  delete headerConstText2; // we dont need that
132 
133  //.bit header vars:
134  fileReadAndDefineBE(headerDesignNameLength, fin, 16);
135  fileReadAndDefineChars(designNameChars, fin, headerDesignNameLength);
136 
137  if(verbose)cout<<"\tName of design: "<<designNameChars<<endl;
138  designName = string(designNameChars);
139  delete designNameChars;
140  for(int headerDone = 0 ; !headerDone ; ){
141  fileReadAndDefineBE(key, fin, 8);
142  switch(key){
143  case 'b': {
144  fileReadAndDefineBE(headerAttrLength, fin, 16);
145  fileReadAndDefineChars(partNameChars, fin, headerAttrLength);
146  if(verbose)cout<<"\tFPGA part: "<<partNameChars<<endl;
147  partName = string(partNameChars);
148  delete partNameChars;
149  break;
150  }
151  case 'c': {
152  fileReadAndDefineBE(headerAttrLength, fin, 16);
153  fileReadAndDefineChars(date, fin, headerAttrLength);
154  if(verbose)cout<<"\tdate: "<<date<<endl;
155  fileDate = string(date);
156  delete date;
157  break;
158  }
159  case 'd': {
160  fileReadAndDefineBE(headerAttrLength, fin, 16);
161  fileReadAndDefineChars(time, fin, headerAttrLength);
162  if(verbose)cout<<"\ttime: "<<time<<endl;
163  fileTime = string(time);
164  delete time;
165  break;
166  }
167  case 'e': {
168  headerDone = 1;//last attribute in a .bit header
169  int reportedRemainingFileLength;
170  fileReadBE(reportedRemainingFileLength, fin, 32);
171  int tmpPos = (int)fin.tellg();
172  fin.seekg (0, fin.end);
173  int fileSize = (int)fin.tellg();
174  fin.seekg (tmpPos, fin.beg);
175  if(warn && fileSize != (tmpPos + reportedRemainingFileLength))
176  cout << "warning: .bit header contained inaccurate file length field.\n";
177  break;
178  }
179  }
180  }
181  //Follow some 0xFF's and the bus width detection c0nstant "\xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF\x44\x00\x22\x11\xBB\x00\x00\x00\xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF"
182  //However we don't care about reading those
183 
184  //Find sync
185  for(int syncDetectionDone = 0 ; !syncDetectionDone ; ){
186  fileReadAndDefineBE(word, fin, 32);
187  if(word == 0xAA995566)
188  syncDetectionDone++;
189  else
190  fin.seekg(-3,ios::cur);
191  }
192 }
