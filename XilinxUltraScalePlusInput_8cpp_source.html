30 void XilinxUltraScalePlus::readBitstream(string filename)
31 {
32  ifstream fin (filename, ifstream::binary);
33  if(!fin.good())
34  throw runtime_error(string("Could not open file: \"").append(filename).append("\" .\n"));
35  log("Reading Xilinx UltraScale+ bitstream from file \"" + filename + "\"");
36 
37  loadedBitstreamEndianess = XilinxUltraScalePlus::parseBitstreamEndianess(fin);
38 
39  if(str::iff::stringEndsWith(filename, ".bit"))
40  XilinxUltraScalePlus::readBitstreamBIT(fin);
41  else if(str::iff::stringEndsWith(filename, ".bin"))
42  XilinxUltraScalePlus::readBitstreamBIN(fin);
43  else
44  throw runtime_error(string("Unknown Xilinx UltraScale+ file format tried to be read.\n"));
45  fin.close();
46 }

66 void XilinxUltraScalePlus::readBitstreamMain(ifstream& fin)
67 {
68  int slr = 0;
69 
70  CAP::Register regAddr = CAP::Register::UNDEFINED;
71  int wordCount = 0;
72  char shadowFrame[XUSP_WORDS_PER_FRAME*4];
73  int shadowFrameValid = 0;
74  int b = 7, r = 0, c = 0, m = 0;
75  //Parse bitstream
76  for( ; ; ){
77  uint32_t instruction = FileIO::read32(fin, loadedBitstreamEndianess);
78  if(!fin.good()){
79  break; // done with the bitstream
80  } else {
81  int instructionType = XilinxUltraScalePlus::CAP_getInstructionType(instruction);
82  CAP::Operation instructionOPCODE = XilinxUltraScalePlus::CAP_getInstructionOperation(instruction);
83  int instructionPayload = XilinxUltraScalePlus::CAP_getInstructionPayload(instruction);
84  if(instructionType == 1) {
85  regAddr = XilinxUltraScalePlus::CAP_getInstructionRegister(instruction);
86  wordCount = XilinxUltraScalePlus::CAP_getInstructionWordCount(instruction);
87  } else if(instructionType == 2) {
88  wordCount = instructionPayload;
89  } else {
90  throw runtime_error("Bitstream has invalid instruction (invalid type).");
91  }
92  if(instructionOPCODE == CAP::Operation::WRITE){ //Write register
93  for( ; wordCount > 0 ; ){
94  if(regAddr == CAP::Register::FAR){
95  uint32_t farValue = FileIO::read32(fin, loadedBitstreamEndianess);
96  wordCount--;
97  XilinxUltraScalePlus::CAP_parseFAR(farValue, slr, b, r, c, m);
98  } else if(regAddr == CAP::Register::FDRI){
99  if(wordCount % XUSP_WORDS_PER_FRAME != 0)
100  throw runtime_error("FDRI write of a partial frame was detected, which is currently not supported.");
101  if(shadowFrameValid){
102  if(b == 0){ // CLB
103  memcpy((char*)&bitstreamCLB[r][c][m*XUSP_WORDS_PER_FRAME] ,&shadowFrame, XUSP_WORDS_PER_FRAME*4);
104  } else if(b == 1) { // bram;
105  memcpy((char*)&bitstreamBRAM[r][c][m*XUSP_WORDS_PER_FRAME] ,&shadowFrame, XUSP_WORDS_PER_FRAME*4);
106  }
107  XilinxUltraScalePlus::CAP_IncrementFAR(slr, b, r, c, m);
108  }
109  int forwardShadowReg = (wordCount/XUSP_WORDS_PER_FRAME) - 1;//skip shadow reg for certain number of frames
110  if(forwardShadowReg > 0){
111  if(b == 0){ // CLB
112  fin.read((char*)&bitstreamCLB[r][c][m*XUSP_WORDS_PER_FRAME], forwardShadowReg * XUSP_WORDS_PER_FRAME * 4);
113  } else if(b == 1){
114  fin.read((char*)&bitstreamBRAM[r][c][m*XUSP_WORDS_PER_FRAME], forwardShadowReg * XUSP_WORDS_PER_FRAME * 4);
115  }
116  for(int i = 0 ; i < forwardShadowReg ; i++){
117  XilinxUltraScalePlus::CAP_IncrementFAR(slr, b, r, c, m);
118  }
119  }
120  wordCount -= forwardShadowReg * XUSP_WORDS_PER_FRAME;
121  shadowFrameValid = 1;
122  fin.read((char*)&shadowFrame, XUSP_WORDS_PER_FRAME*4);
123  wordCount -= XUSP_WORDS_PER_FRAME;
124 
125  } else if(regAddr == CAP::Register::CMD){
126  CAP::Command command = static_cast<CAP::Command>(FileIO::read32(fin, loadedBitstreamEndianess));
127  wordCount--;
128  if(command == CAP::Command::WCFG){
129  shadowFrameValid = 0;
130  }
131  } else if(regAddr == CAP::Register::MAGIC1){
132  slr++;
133  wordCount = 0;
134  shadowFrameValid = 0;
135  } else {
136  uint32_t scrap = FileIO::read32(fin, loadedBitstreamEndianess);
137  wordCount--;
138  }
139  }//Write register count for loop
140  }//Write register instruction
141  }// not end of file
142  }// for loop parsing the rest of the bitstream
143 }
