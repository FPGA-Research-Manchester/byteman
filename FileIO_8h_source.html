326  {
327  uint8_t nullChar = 0;

329  for( ; (0 != writeCharArray[charArraySize-1] && charArraySize != outputSize); charArraySize++);
330  fout.write(writeCharArray, charArraySize);

346  inline void writeBitSwappedCharArray(ofstream& fout, char* writeCharArray, int outputSize = 0)
347  {
348  uint8_t nextChar = 0;
349  int charArraySize = 1;
350  for( ; (0 != writeCharArray[charArraySize-1] && charArraySize != outputSize); charArraySize++);
351  for(int i = 0 ; i < charArraySize ; i++){
352  nextChar = Endian::BitSwap8(writeCharArray[i]);
353  fout.write(reinterpret_cast<char*>(&nextChar), 1);
354  }
355  nextChar = 0;

357  fout.write(reinterpret_cast<char*>(&nextChar), 1);

359  inline void writeBitSwappedString(ofstream& fout, string writeString, int outputSize = 0)
360  {
361  uint8_t nextChar = 0;
362  int stringSize = writeString.size() + 1;
363  if((0 == outputSize) || (outputSize >= stringSize)) {
364  for(int i = 0 ; i < stringSize ; i++){
365  nextChar = Endian::BitSwap8(writeString.at(i));
366  fout.write(reinterpret_cast<char*>(&nextChar), 1);
367  }
368  } else {
369  for(int i = 0 ; i < outputSize ; i++){
370  nextChar = Endian::BitSwap8(writeString.at(i));
371  fout.write(reinterpret_cast<char*>(&nextChar), 1);
372  }
373  }
