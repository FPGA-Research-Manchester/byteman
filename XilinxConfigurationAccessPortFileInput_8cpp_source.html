101  //Follow some 0xFF's and the bus width detection c0nstant "\xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF\x44\x00\x22\x11\xBB\x00\x00\x00\xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF\xFF"
102  //However we don't care about reading those

104  //Find sync
105  for(int syncDetectionDone = 0 ; !syncDetectionDone ; ){

113  returnVal = Endianess::BE;

116  returnVal = Endianess::LE;

118  } else if(XCAP_SyncInstruction() == Endian::BitSwap32(Endian::NativeToBigEndian32(word))){
119  returnVal = Endianess::BE_BS;

121  } else if(XCAP_SyncInstruction() == Endian::BitSwap32(Endian::NativeToLittleEndian32(word))){

126  }
