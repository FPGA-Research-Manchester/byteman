32 void XilinxUltraScalePlus::assembler(string filenameIn, string filenameOut){
33  enum FILEFORMAT {FILE_NULL = 0, FILE_BIT, FILE_BIT_ASM};
34  FILEFORMAT fileformatIn = FILE_NULL, fileformatOut = FILE_NULL;
35 
36  if(fileExtensionIs(filenameIn, "bit"))
37  fileformatIn = FILE_BIT;
38  if(fileExtensionIs(filenameIn, "bitasm"))
39  fileformatIn = FILE_BIT_ASM;
40 
41  if(fileExtensionIs(filenameOut, "bit"))
42  fileformatOut = FILE_BIT;
43  if(fileExtensionIs(filenameOut, "bitasm"))
44  fileformatOut = FILE_BIT_ASM;
45  if(fileformatIn == FILE_NULL)
46  throw runtime_error(string("Unknown Xilinx UltraScale+ file format tried to be read by the assembler. See \"byteman -h assembly\".\n"));
47  if(fileformatOut == FILE_NULL)
48  throw runtime_error(string("Unknown Xilinx UltraScale+ file format tried to be written by the assembler. See \"byteman -h assembly\".\n"));
49  if(fileformatIn == fileformatOut)
50  throw runtime_error(string("Unknown Xilinx UltraScale+ assembler operation between identical file formats. See \"byteman -h assembly\".\n"));
51 
52 
53  ifstream fin (filenameIn, ifstream::binary);
54  if(!fin.good())
55  throw runtime_error(string("Could not open file: \"").append(filenameIn).append("\" .\n"));
56 
57  ofstream fout (filenameOut, ofstream::binary | ofstream::trunc);
58  if(!fout.good())
59  throw runtime_error(string("Could not open file: \"").append(filenameOut).append("\"!\n"));
60 
61  if(fileformatIn == FILE_BIT && fileformatOut == FILE_BIT_ASM)
62  XilinxUltraScalePlus::disassemblerBitToAsm(fin, fout);
63  if(fileformatIn == FILE_BIT_ASM && fileformatOut == FILE_BIT)
64  XilinxUltraScalePlus::assemblerAsmToBit(fin, fout);
65 
66  fin.close();
67  fout.close();
68 }

120 void XilinxUltraScalePlus::assemblerAsmToBit(ifstream& fin, ofstream& fout){
121 
122  for (string line; getline(fin, line); ) {
123  auto firstEqPos = line.find_first_of('=');
124  if(firstEqPos != string::npos)
125  line.replace(firstEqPos, 1, '=', ' ');
126  #define has(x) (line.find(x) != string::npos)
127  if(has("Name"))designName = parseString::removeQuotes(parseString::getLastWord(line));
128  if(has("FPGA"))partName = parseString::removeQuotes(parseString::getLastWord(line));
129  if(has("Date"))fileDate = parseString::removeQuotes(parseString::getLastWord(line));
130  if(has("Time"))fileTime = parseString::removeQuotes(parseString::getLastWord(line));
131  if(has("HEADER END"))break;
132  #undef has
133  }
134  XilinxUltraScalePlus::setDevice(XilinxUltraScalePlus::getDeviceByNameOrThrow(partName), partName);
135  XilinxUltraScalePlus::initializeResourceStringParameters();
136  streamoff headerLocationOfRemainingFileLength = XilinxUltraScalePlus::outputBITheader(fout);
137 
138  int regAddr = XUSP_CAP_REG_UNKNOWN;
139  int frameCount = 0;
140  int b = 7, r = 0, c = 0, m = 0;
141  for (string line; getline(fin, line); ) {
142  #define has(x) (line.find(x) != string::npos)
143  #define hOR(x,y) (has(x)||has(y))
144  if(line.at(line.find_first_not_of(" \t")) == '#')// if #, skip that line as comment
145  continue;
146  transform(line.begin(), line.end(),line.begin(), ::toupper);
147  replace(line.begin(), line.end(), '=', ' ');
148  replace(line.begin(), line.end(), '#', ' ');
149  replace(line.begin(), line.end(), ',', ' ');
150  if(has("NOP")){
151  int nopHiddenValue;
152  if(!parseString::getMultipleInts(line, nopHiddenValue))nopHiddenValue = 0;
153  CAPwriteNOPwData(fout, nopHiddenValue);
154  }
155  else if(has("RESERVED")){
156  int reservedHiddenValue;
157  if(!parseString::getMultipleInts(line, reservedHiddenValue))reservedHiddenValue = 0;
158  CAPwriteRESERVEDwData(fout, reservedHiddenValue);
159  }
160  else if(has("@")){
161  int newRegAddr = assemblerGetRegID(line);
162  if(has("READ REG @")){
163  int readLength;
164  if(!parseString::getMultipleInts(line, readLength)) readLength = 1; // default read length is 1 if unspecified
165  CAPreadReg(fout, newRegAddr, readLength);
166  } else {// must be a write then ¯\_(ツ)_/¯
167 
168  if(newRegAddr == XUSP_CAP_REG_UNKNOWN){
169  throw runtime_error(string("Couldn't parse assembly command: \"").append(line).append("\"!"));
170  } else if(has("SELECT REGISTER")){
171  CAPselectReg(fout, newRegAddr);
172  } else if(newRegAddr == XUSP_CAP_REG_FDRI){
173  if(!parseString::getMultipleInts(line, frameCount))
174  throw runtime_error(string("FDRI command needs size: \"").append(line).append("\"!"));
175  int wordCount = frameCount * XUSP_WORDS_PER_FRAME;
176  if(regAddr == XUSP_CAP_REG_FDRI) {
177  CAPwriteFDRI2(fout, wordCount);
178  } else {
179  CAPwriteFDRI1(fout, wordCount);
180  }
181  string frameLine;
182  for (string frameLine; getline(fin, frameLine); ) {
183  if(frameLine.at(frameLine.find_first_not_of(" \t")) == '#')// if #, skip that line as comment
184  continue;
185  uint32_t frameData[XUSP_WORDS_PER_FRAME];
186  if(!parseString::getArrayOfUints(frameLine, XUSP_WORDS_PER_FRAME, &frameData[0]))
187  throw runtime_error(string("Was expecting the data for a full frame on this line: \"").append(frameLine).append("\", because I think there are ").append(to_string(frameCount)).append(" frames left."));
188  //first 3 words are clk, next 90 are the 0-45 and 48-93 data words
189  for (int w = XUSP_WORDS_AT_CLK; w < (XUSP_WORDS_AT_CLK + XUSP_WORDS_BEFORE_CLK) ; w++){
190  fileWriteBE(fout, 32, frameData[w]);
191  }
192  for (int w = 0; w < XUSP_WORDS_AT_CLK ; w++){
193  fileWriteBE(fout, 32, frameData[w]);
194  }
195  for (int w = (XUSP_WORDS_AT_CLK + XUSP_WORDS_BEFORE_CLK); w < XUSP_WORDS_PER_FRAME ; w++){
196  fileWriteBE(fout, 32, frameData[w]);
197  }
198 
199  frameCount--;
200  if(frameCount == 0)
201  break;
202  }
203  if(frameCount != 0)
204  throw runtime_error(string("End of file reached while performing FDRI frame writes.!"));
205  } else if(newRegAddr == XUSP_CAP_REG_FAR){
206  if(!parseString::getMultipleInts(line, b, r, c, m))
207  throw runtime_error(string("Could not parse the new FAR value: \"").append(line).append("\"!"));
208  CAPwriteFAR(fout, b, r, c, m);
209  } else {
210  int newValue;
211  if(!parseString::getMultipleInts(line, newValue))
212  throw runtime_error(string("Could not parse the new register value: \"").append(line).append("\"!"));
213  CAPwriteReg(fout, newRegAddr, newValue);
214  }
215  }
216  regAddr = newRegAddr;
217  } else {// ! has("@")
218  //must have been a command then
219  int cmdID = assemblerGetCommandID(line);
220  if(cmdID == XUSP_CAP_CMD_UNKNOWN)
221  throw runtime_error(string("Couldn't parse assembly command: \"").append(line).append("\"!"));
222  CAPwriteReg(fout, XUSP_CAP_REG_CMD, cmdID);
223  }
224  #undef has
225  #undef hOR
226  }
227  XilinxUltraScalePlus::outputBITheaderLengthField(fout, headerLocationOfRemainingFileLength);
228 }

279 void XilinxUltraScalePlus::disassemblerBitToAsm(ifstream& fin, ofstream& fout){
280  XilinxUltraScalePlus::readBITheader(fin);
281  XilinxUltraScalePlus::setDevice(XilinxUltraScalePlus::getDeviceByNameOrThrow(partName), partName);
282  XilinxUltraScalePlus::initializeResourceStringParameters();
283  XilinxUltraScalePlus::disassemblerWriteHeader(fout);
284 
285  int regAddr = XUSP_CAP_REG_UNKNOWN;
286  int wordCount = 0;
287  int shadowFrameValid = 0;
288  int b = 7, r = 0, c = 0, m = 0;
289  //Parse bitstream
290  for( ; ; ){
291  fileReadAndDefineBE(instruction, fin, 32);
292  if(!fin.good()){
293  break; // done with the bitstream
294  } else {
295  int instructionType = getInstructionType(instruction);
296  int instructionOPCODE = getInstructionOPCODE(instruction);
297  int instructionPayload = getInstructionPayload(instruction);
298  if(instructionType == 1) {
299  wordCount = getInstructionType1WordCount(instruction);
300  regAddr = getInstructionType1Register(instruction);
301  } else if(instructionType == 2) {
302  wordCount = instructionPayload;
303  } else {
304  fout << "Invalid instruction (invalid type): 0x" << hex << setw(8) << setfill('0') << instruction << endl;
305  }
306 
307  if(instructionOPCODE == OP_NOP) {
308  if(instructionPayload != 0)
309  fout << "NOP #" << instructionPayload << endl;
310  else
311  fout << "NOP" << endl;
312  } else if(instructionOPCODE == OP_RESERVED) {
313  if(instructionPayload != 0)
314  fout << "RESERVED #" << instructionPayload << endl;
315  else
316  fout << "RESERVED" << endl;
317  } else if(instructionOPCODE == OP_READ) {
318  fout << "Read Reg @";
319  writeRegisterName(fout, regAddr);
320  fout << " for length #" << wordCount << endl;
321  } else { // OP_WRITE
322  if((regAddr == XUSP_CAP_REG_FDRI) && (wordCount > 0) && (wordCount % XUSP_WORDS_PER_FRAME == 0)) {
323  if(shadowFrameValid) {
324  fout<<"# Previous shadow register contents will be written to address (BlockType="<<b<<", RowAddress="<<r<<", MajorAddress="<<c<<", MinorAddress"<<m<<endl<<") hex data:"<<endl;
325  XUSP_INCR_FAR(b,r,c,m)
326  }
327  shadowFrameValid = 1;
328  int frameCount = (wordCount/XUSP_WORDS_PER_FRAME);
329  fout << dec << "@FDRI for #" << frameCount << " frames:" << endl;
330  for(int i = 0 ; i < frameCount ; i++){
331  fout << "# ";
332  if(i == (frameCount-1)) fout << "(This is written to shadow register)";
333  fout << dec << "Writing frame #" << i << " located at address (BlockType="<<b<<", RowAddress="<<r<<", MajorAddress="<<c<<", MinorAddress="<<m<<") hex data:"<<endl;
334  uint32_t frameData[XUSP_WORDS_PER_FRAME];
335  for(int w = 0 ; w < XUSP_WORDS_PER_FRAME ; w++){
336  fileReadBE(frameData[w], fin, 32);
337  }
338  fout<<"CLOCK: ";
339  for(int w = XUSP_WORDS_BEFORE_CLK ; w < (XUSP_WORDS_BEFORE_CLK + XUSP_WORDS_AT_CLK) ; w++){
340  fout<< "0x" << hex << setw(8) << setfill('0') << frameData[w] << " ";
341  }
342  fout<<" ;DATA: ";
343  for(int w = 0 ; w < XUSP_WORDS_BEFORE_CLK ; w++){
344  fout<< "0x" << hex << setw(8) << setfill('0') << frameData[w] << " ";
345  }
346  for(int w = (XUSP_WORDS_BEFORE_CLK + XUSP_WORDS_AT_CLK) ; w < XUSP_WORDS_PER_FRAME ; w++){
347  fout<< "0x" << hex << setw(8) << setfill('0') << frameData[w] << " ";
348  }
349  fout << endl;
350  XUSP_INCR_FAR(b,r,c,m)
351  }
352  } else if(regAddr == XUSP_CAP_REG_CMD && wordCount == 1){
353  fileReadAndDefineBE(writeData, fin, 32);
354  writeCommandName(fout, writeData);
355  fout << ";";
356  if(writeData == XUSP_CAP_CMD_WCFG){
357  shadowFrameValid = 0;
358  fout << " (also clears shadow register)";
359  }
360  fout << endl;
361  } else if((instructionType == 1) && (wordCount == 0)){
362  fout << "Select register @";
363  writeRegisterName(fout, regAddr);
364  fout << endl;
365  } else if(wordCount == 1){
366  fileReadAndDefineBE(writeData, fin, 32);
367  fout << "@";
368  writeRegisterName(fout, regAddr);
369  if(regAddr == XUSP_CAP_REG_FAR) {
370  parseFAR(writeData,b,r,c,m);
371  fout << " = BlockType=" << dec <<b<<" RowAddress="<<r<<" MajorAddress="<<c<<" MinorAddress="<<m<< endl;
372  } else {
373  fout << " = 0x" << hex << setw(8) << setfill('0') << writeData << endl;
374  }
375  } else {
376  fout << "Bad instruction: 0x"<< hex << setw(8) << setfill('0') << instruction << endl;
377  }
378  } // OP_WRITE
379  } // if fin is still good
380  } // for(;;)
