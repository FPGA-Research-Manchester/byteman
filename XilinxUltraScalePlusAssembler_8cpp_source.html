38 void XilinxUltraScalePlus::assembler(string filenameIn, string filenameOut)
39 {
40  enum FILEFORMAT {FILE_NULL = 0, FILE_BIT, FILE_BIT_ASM};
41  FILEFORMAT fileformatIn = FILE_NULL, fileformatOut = FILE_NULL;
42 
43  if(StringFuncs::checkIf::stringEndsWith(filenameIn, ".bit"))
44  fileformatIn = FILE_BIT;
45  if(StringFuncs::checkIf::stringEndsWith(filenameIn, ".bitasm"))
46  fileformatIn = FILE_BIT_ASM;
47 
48  if(StringFuncs::checkIf::stringEndsWith(filenameOut, ".bit"))
49  fileformatOut = FILE_BIT;
50  if(StringFuncs::checkIf::stringEndsWith(filenameOut, ".bitasm"))
51  fileformatOut = FILE_BIT_ASM;
52  if(fileformatIn == FILE_NULL)
53  throw runtime_error(string("Unknown Xilinx UltraScale+ file format tried to be read by the assembler. See \"byteman -h assembly\".\n"));
54  if(fileformatOut == FILE_NULL)
55  throw runtime_error(string("Unknown Xilinx UltraScale+ file format tried to be written by the assembler. See \"byteman -h assembly\".\n"));
56  if(fileformatIn == fileformatOut)
57  throw runtime_error(string("Unknown Xilinx UltraScale+ assembler operation between identical file formats. See \"byteman -h assembly\".\n"));
58 
59 
60  ifstream fin (filenameIn, ifstream::binary);
61  if(!fin.good())
62  throw runtime_error(string("Could not open file: \"").append(filenameIn).append("\" .\n"));
63 
64  ofstream fout (filenameOut, ofstream::binary | ofstream::trunc);
65  if(!fout.good())
66  throw runtime_error(string("Could not open file: \"").append(filenameOut).append("\"!\n"));
67 
68  if(fileformatIn == FILE_BIT && fileformatOut == FILE_BIT_ASM)
69  XilinxUltraScalePlus::disassemblerBitToAsm(fin, fout);
70  if(fileformatIn == FILE_BIT_ASM && fileformatOut == FILE_BIT)
71  XilinxUltraScalePlus::assemblerAsmToBit(fin, fout);
72 
73  fin.close();
74  fout.close();
75 }

129 void XilinxUltraScalePlus::assemblerAsmToBit(ifstream& fin, ofstream& fout)
130 {
131  for (string line; getline(fin, line); ) {
132  auto firstEqPos = line.find_first_of('=');
133  if(firstEqPos != string::npos)
134  line.replace(firstEqPos, 1, '=', ' ');
135  #define has(x) (line.find(x) != string::npos)
136  if(has("Name"))designName = StringFuncs::removeExternalQuotes(StringFuncs::parse::lastStringWord(line));
137  if(has("FPGA"))partName = StringFuncs::removeExternalQuotes(StringFuncs::parse::lastStringWord(line));
138  if(has("Date"))fileDate = StringFuncs::removeExternalQuotes(StringFuncs::parse::lastStringWord(line));
139  if(has("Time"))fileTime = StringFuncs::removeExternalQuotes(StringFuncs::parse::lastStringWord(line));
140  if(has("HEADER END"))break;
141  #undef has
142  }
143  XilinxUltraScalePlus::setDevice(XilinxUltraScalePlus::getDeviceByNameOrThrow(partName), partName);
144  XilinxUltraScalePlus::initializeResourceStringParameters();
145  streamoff headerLocationOfRemainingFileLength = XilinxUltraScalePlus::outputBITheader(fout, Endianess::BE);
146 
147  CAP::Register regAddr = CAP::Register::UNDEFINED;
148  int frameCount = 0;
149  int b = 7, r = 0, c = 0, m = 0;
150  for (string line; getline(fin, line); ) {
151  #define has(x) (line.find(x) != string::npos)
152  #define hOR(x,y) (has(x)||has(y))
153  if(line.at(line.find_first_not_of(" \t")) == '#')// if #, skip that line as comment
154  continue;
155  transform(line.begin(), line.end(),line.begin(), ::toupper);
156  replace(line.begin(), line.end(), '=', ' ');
157  replace(line.begin(), line.end(), '#', ' ');
158  replace(line.begin(), line.end(), ',', ' ');
159  if(has("NOP")){
160  int nopHiddenValue;
161  if(!StringFuncs::parse::multipleInts(line, nopHiddenValue))nopHiddenValue = 0;
162  XilinxUltraScalePlus::CAP_writeNOP(fout, 1, nopHiddenValue, Endianess::BE);
163  }
164  else if(has("RESERVED")){
165  int reservedHiddenValue;
166  if(!StringFuncs::parse::multipleInts(line, reservedHiddenValue))reservedHiddenValue = 0;
167  XilinxUltraScalePlus::CAP_writeRESERVED(fout, 1, reservedHiddenValue, Endianess::BE);
168  }
169  else if(has("@")){
170  CAP::Register newRegAddr = assemblerGetRegID(line);
171  if(has("READ REG @")){
172  int readLength;
173  if(!StringFuncs::parse::multipleInts(line, readLength)) readLength = 1; // default read length is 1 if unspecified
174  XilinxUltraScalePlus::CAP_writeReadRegister(fout, newRegAddr, readLength, Endianess::BE);
175  } else {// must be a write then \_()_/
176 
177  if(newRegAddr == CAP::Register::UNDEFINED){
178  throw runtime_error(string("Couldn't parse assembly command: \"").append(line).append("\"!"));
179  } else if(has("SELECT REGISTER")){
180  XilinxUltraScalePlus::CAP_writeSelectRegister(fout, newRegAddr, Endianess::BE);
181  } else if(newRegAddr == CAP::Register::FDRI){
182  if(!StringFuncs::parse::multipleInts(line, frameCount))
183  throw runtime_error(string("FDRI command needs size: \"").append(line).append("\"!"));
184  int wordCount = frameCount * XUSP_WORDS_PER_FRAME;
185  if(regAddr == CAP::Register::FDRI) {
186  XilinxUltraScalePlus::CAP_writeFDRI2(fout, wordCount, Endianess::BE);
187  } else {
188  XilinxUltraScalePlus::CAP_writeFDRI1(fout, wordCount, Endianess::BE);
189  }
190  if(frameCount > 0){
191  string frameLine;
192  for (string frameLine; getline(fin, frameLine); ) {
193  if(frameLine.at(frameLine.find_first_not_of(" \t")) == '#')// if #, skip that line as comment
194  continue;
195  uint32_t frameData[XUSP_WORDS_PER_FRAME];
196  if(!StringFuncs::parse::arrayOfUints(frameLine, XUSP_WORDS_PER_FRAME, &frameData[0]))
197  throw runtime_error(string("Was expecting the data for a full frame on this line: \"").append(frameLine).append("\", because I think there are ").append(to_string(frameCount)).append(" frames left."));
198  //first 3 words are clk, next 90 are the 0-45 and 48-93 data words
199  for (int w = XUSP_WORDS_AT_CLK; w < (XUSP_WORDS_AT_CLK + XUSP_WORDS_BEFORE_CLK) ; w++){
200  FileIO::write32(fout, frameData[w], Endianess::BE);
201  }
202  for (int w = 0; w < XUSP_WORDS_AT_CLK ; w++){
203  FileIO::write32(fout, frameData[w], Endianess::BE);
204  }
205  for (int w = (XUSP_WORDS_AT_CLK + XUSP_WORDS_BEFORE_CLK); w < XUSP_WORDS_PER_FRAME ; w++){
206  FileIO::write32(fout, frameData[w], Endianess::BE);
207  }
208 
209  frameCount--;
210  if(frameCount == 0)
211  break;
212  }
213  if(frameCount != 0)
214  throw runtime_error(string("End of file reached while performing FDRI frame writes.!"));
215  }
216  } else if(newRegAddr == CAP::Register::FAR){
217  if(!StringFuncs::parse::multipleInts(line, b, r, c, m))
218  throw runtime_error(string("Could not parse the new FAR value: \"").append(line).append("\"!"));
219  uint32_t farValue = XilinxUltraScalePlus::CAP_makeFAR(b, r, c, m);
220  XilinxUltraScalePlus::CAP_writeRegister(fout, CAP::Register::FAR, farValue, Endianess::BE);
221  } else {
222  int newValue;
223  if(!StringFuncs::parse::multipleInts(line, newValue))
224  throw runtime_error(string("Could not parse the new register value: \"").append(line).append("\"!"));
225  XilinxUltraScalePlus::CAP_writeRegister(fout, newRegAddr, newValue, Endianess::BE);
226  }
227  }
228  regAddr = newRegAddr;
229  } else {// ! has("@")
230  //must have been a command then
231  CAP::Command cmdID = assemblerGetCommandID(line);
232  if(cmdID == CAP::Command::UNDEFINED)
233  throw runtime_error(string("Couldn't parse assembly command: \"").append(line).append("\"!"));
234  XilinxUltraScalePlus::CAP_writeCommand(fout, cmdID, Endianess::BE);
235  }
236  #undef has
237  #undef hOR
238  }
239  XilinxUltraScalePlus::outputBITheaderLengthField(fout, headerLocationOfRemainingFileLength, Endianess::BE);
240 }

293 void XilinxUltraScalePlus::disassemblerBitToAsm(ifstream& fin, ofstream& fout)
294 {
295  loadedBitstreamEndianess = XilinxUltraScalePlus::parseBitstreamEndianess(fin);
296  XilinxUltraScalePlus::parseBITheader(fin, loadedBitstreamEndianess);
297  XilinxUltraScalePlus::setDevice(XilinxUltraScalePlus::getDeviceByNameOrThrow(partName), partName);
298  XilinxUltraScalePlus::initializeResourceStringParameters();
299  XilinxUltraScalePlus::disassemblerWriteHeader(fout);
300 
301  CAP::Register regAddr = CAP::Register::UNDEFINED;
302  int wordCount = 0;
303  int shadowFrameValid = 0;
304  int b = 7, r = 0, c = 0, m = 0;
305  //Parse bitstream
306  for( ; ; ){
307  uint32_t instruction = FileIO::read32(fin, Endianess::BE);
308  if(!fin.good()){
309  break; // done with the bitstream
310  } else {
311  int instructionType = XilinxUltraScalePlus::CAP_getInstructionType(instruction);
312  CAP::Operation instructionOPCODE = XilinxUltraScalePlus::CAP_getInstructionOperation(instruction);
313  int instructionPayload = XilinxUltraScalePlus::CAP_getInstructionPayload(instruction);
314  if(instructionType == 1) {
315  wordCount = XilinxUltraScalePlus::CAP_getInstructionWordCount(instruction);
316  regAddr = XilinxUltraScalePlus::CAP_getInstructionRegister(instruction);
317  } else if(instructionType == 2) {
318  wordCount = instructionPayload;
319  } else {
320  fout << "0x" << hex << setw(8) << setfill('0') << instruction << " (Invalid instruction [invalid type])" << endl;
321  continue;
322  }
323 
324  if(instructionOPCODE == CAP::Operation::NOP) {
325  if(instructionPayload != 0)
326  fout << "NOP #" << instructionPayload << endl;
327  else
328  fout << "NOP" << endl;
329  } else if(instructionOPCODE == CAP::Operation::RESERVED) {
330  if(instructionPayload != 0)
331  fout << "RESERVED #" << instructionPayload << endl;
332  else
333  fout << "RESERVED" << endl;
334  } else if(instructionOPCODE == CAP::Operation::READ) {
335  fout << "Read Reg @";
336  writeRegisterName(fout, regAddr);
337  fout << " for length #" << wordCount << endl;
338  } else { // CAP::Operation::WRITE
339  if((regAddr == CAP::Register::FDRI) && (wordCount > 0) && (wordCount % XUSP_WORDS_PER_FRAME == 0)) {
340  if(shadowFrameValid) {
341  fout << "# Shadow register contents are written to frame (BlockType="<<b<<", RowAddress="<<r<<", MajorAddress="<<c<<", MinorAddress"<<m<<") (Frame type: " << XilinxUltraScalePlus::getFrameType(b,r,c) << ")."<<endl;
342  XilinxUltraScalePlus::CAP_IncrementFAR(b, r, c, m);
343  }
344  shadowFrameValid = 1;
345  int frameCount = (wordCount/XUSP_WORDS_PER_FRAME);
346  fout << dec << "@FDRI for #" << frameCount << " frames:" << endl;
347  for(int i = 0 ; i < frameCount ; i++){
348  fout << "# ";
349  if(i == (frameCount-1)) fout << "(This frame data is written to shadow register!)";
350  fout << dec << "Writing frame #" << i << " (BlockType="<<b<<", RowAddress="<<r<<", MajorAddress="<<c<<", MinorAddress="<<m<<") (Frame type: " << XilinxUltraScalePlus::getFrameType(b,r,c) << ") hex data:"<<endl;
351  uint32_t frameData[XUSP_WORDS_PER_FRAME];
352  for(int w = 0 ; w < XUSP_WORDS_PER_FRAME ; w++){
353  frameData[w] = FileIO::read32(fin, Endianess::BE);
354  }
355  fout<<"CLOCK: ";
356  for(int w = XUSP_WORDS_BEFORE_CLK ; w < (XUSP_WORDS_BEFORE_CLK + XUSP_WORDS_AT_CLK) ; w++){
357  fout<< "0x" << hex << setw(8) << setfill('0') << frameData[w] << " ";
358  }
359  fout<<" ;DATA: ";
360  for(int w = 0 ; w < XUSP_WORDS_BEFORE_CLK ; w++){
361  fout<< "0x" << hex << setw(8) << setfill('0') << frameData[w] << " ";
362  }
363  for(int w = (XUSP_WORDS_BEFORE_CLK + XUSP_WORDS_AT_CLK) ; w < XUSP_WORDS_PER_FRAME ; w++){
364  fout<< "0x" << hex << setw(8) << setfill('0') << frameData[w] << " ";
365  }
366  fout << endl;
367  XilinxUltraScalePlus::CAP_IncrementFAR(b, r, c, m);
368  }
369  } else if(regAddr == CAP::Register::CMD && wordCount == 1){
370  uint32_t writeData = FileIO::read32(fin, Endianess::BE);
371  writeCommandName(fout, static_cast<CAP::Command>(writeData));
372  fout << ";";
373  if(CAP::Command::WCFG == static_cast<CAP::Command>(writeData)){
374  shadowFrameValid = 0;
375  fout << " (also clears shadow register)";
376  }
377  fout << endl;
378  } else if((instructionType == 1) && (wordCount == 0)){
379  fout << "Select register @";
380  writeRegisterName(fout, regAddr);
381  fout << endl;
382  } else if(wordCount == 1){
383  uint32_t writeData = FileIO::read32(fin, Endianess::BE);
384  fout << "@";
385  writeRegisterName(fout, regAddr);
386  if(regAddr == CAP::Register::FAR) {
387  XilinxUltraScalePlus::CAP_parseFAR(writeData, b, r, c, m);
388  fout << " = BlockType=" << dec <<b<<" RowAddress="<<r<<" MajorAddress="<<c<<" MinorAddress="<<m<< endl;
389  } else {
390  fout << " = 0x" << hex << setw(8) << setfill('0') << writeData << endl;
391  }
392  } else {
393  fout << "0x"<< hex << setw(8) << setfill('0') << instruction << "(Bad instruction)" << endl;
394  }
395  } // OP_WRITE
396  } // if fin is still good
397  } // for(;;)
398 }
